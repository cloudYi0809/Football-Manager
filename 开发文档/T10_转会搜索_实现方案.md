## 三、搜索服务

```kotlin
// search/PlayerSearchService.kt
class PlayerSearchService(
    private val saveDb: SaveDatabase,
    private val historyDb: HistoryDatabase,
    private val cacheDb: CacheDatabase,
    private val economyService: EconomyService,
    private val config: TransferConfig
) {

    /**
     * V0.1 09 §二.1 搜索
     * 14 项筛选条件
     */
    suspend fun search(
        saveId: Int,
        filter: SearchFilter,
        page: Int = 0,
        pageSize: Int = 50
    ): List<SearchResult> {
        val startTime = System.currentTimeMillis()

        // 1. 优先使用 cache.db 的搜索索引（性能优化）
        val candidates = if (isSimpleFilter(filter)) {
            searchFromCache(filter, page, pageSize)
        } else {
            searchFromDb(saveId, filter, page, pageSize)
        }

        // 2. 计算身价和工资（实时）
        val results = candidates.map { player ->
            val marketValue = economyService.estimateMarketValue(player)
            val expectedWage = economyService.calculateExpectedWage(player)
            val transferStatus = determineTransferStatus(player, saveId)
            val reportLevel = getScoutingReportLevel(player.playerId, saveId)
            val isWatched = isOnWatchlist(player.playerId, saveId)

            SearchResult(
                playerId = player.playerId,
                playerName = getPlayerName(player.playerId),
                age = calculateAge(player.birthDate),
                nationality = player.nationality,
                position = player.primaryPosition,
                currentCa = player.currentCa,
                potentialPa = player.potentialPa,
                clubId = player.currentClubId,
                clubName = player.currentClubId?.let { getClubName(it) },
                leagueId = player.currentClubId?.let { getClubLeagueId(it) },
                marketValue = marketValue,
                wage = expectedWage,
                contractUntil = player.contractUntil,
                transferStatus = transferStatus,
                scoutingReportLevel = reportLevel,
                isOnWatchlist = isWatched
            )
        }

        // 3. 排序
        val sorted = SearchSorter().sort(results, filter.sortBy, filter.sortOrder)

        // 4. 性能监控
        val duration = System.currentTimeMillis() - startTime
        if (duration > config.searchTimeoutMs) {
            log.warn("搜索超时：${duration}ms")
        }

        return sorted
    }

    /**
     * 简单筛选走 cache.db 索引
     */
    private fun isSimpleFilter(filter: SearchFilter): Boolean {
        return filter.name == null &&
               filter.nationalities.isEmpty() &&
               filter.clubIds.isEmpty() &&
               filter.leagueIds.isEmpty()
    }

    private suspend fun searchFromCache(
        filter: SearchFilter,
        page: Int,
        pageSize: Int
    ): List<SavePlayerStateWithAttributes> {
        return cacheDb.playerSearchDao().search(
            positions = filter.positions,
            caMin = filter.caRange?.first,
            caMax = filter.caRange?.last,
            ageMin = filter.ageRange?.first,
            ageMax = filter.ageRange?.last,
            maxValue = filter.maxMarketValue,
            offset = page * pageSize,
            limit = pageSize
        )
    }

    private suspend fun searchFromDb(
        saveId: Int,
        filter: SearchFilter,
        page: Int,
        pageSize: Int
    ): List<SavePlayerStateWithAttributes> {
        return saveDb.savePlayerStateDao().searchWithFilter(
            saveId = saveId,
            name = filter.name,
            positions = filter.positions,
            caMin = filter.caRange?.first,
            caMax = filter.caRange?.last,
            ageMin = filter.ageRange?.first,
            ageMax = filter.ageRange?.last,
            nationalities = filter.nationalities,
            clubIds = filter.clubIds,
            offset = page * pageSize,
            limit = pageSize
        )
    }

    private fun determineTransferStatus(
        player: SavePlayerStateWithAttributes,
        saveId: Int
    ): TransferStatus {
        if (player.currentClubId == null) return TransferStatus.FREE_AGENT
        // 检查是否挂牌
        val isListed = saveDb.saveTransferOfferDao()
            .isPlayerListed(saveId, player.playerId)
        return if (isListed) TransferStatus.TRANSFERABLE else TransferStatus.TRANSFERABLE
    }
}
```

```kotlin
// compare/PlayerCompareService.kt
class PlayerCompareService(
    private val saveDb: SaveDatabase,
    private val economyService: EconomyService
) {

    /**
     * V0.1 03 阵容页：球员对比功能
     * 2-3 名球员属性对比
     */
    suspend fun compare(playerIds: List<Int>, saveId: Int): CompareResult {
        require(playerIds.size in 2..3) { "对比球员数量必须 2-3 人" }

        val players = playerIds.map { id ->
            saveDb.savePlayerStateDao().getByIdWithAttributes(saveId, id)
        }

        return CompareResult(
            players = players.map { player ->
                PlayerCompareData(
                    playerId = player.playerId,
                    playerName = getPlayerName(player.playerId),
                    age = calculateAge(player.birthDate),
                    position = player.primaryPosition,
                    currentCa = player.currentCa,
                    potentialPa = player.potentialPa,
                    marketValue = economyService.estimateMarketValue(player),
                    wage = economyService.calculateExpectedWage(player),
                    attributes = player.attributes.toMap(),
                    contractUntil = player.contractUntil
                )
            },
            bestInCategory = calculateBestInCategory(players)
        )
    }

    private fun calculateBestInCategory(players: List<SavePlayerStateWithAttributes>): Map<String, Int> {
        // 每个属性谁最强，返回 playerId
        val result = mutableMapOf<String, Int>()
        val firstPlayer = players.first()
        firstPlayer.attributes.javaClass.declaredFields.forEach { field ->
            field.isAccessible = true
            val attributeName = field.name
            val bestPlayer = players.maxByOrNull { p ->
                p.attributes.javaClass.getDeclaredField(attributeName)
                    .apply { isAccessible = true }
                    .getInt(p.attributes)
            }
            result[attributeName] = bestPlayer?.playerId ?: -1
        }
        return result
    }
}

data class CompareResult(
    val players: List<PlayerCompareData>,
    val bestInCategory: Map<String, Int>   // 属性名 -> 最佳 playerId
)

data class PlayerCompareData(
    val playerId: Int,
    val playerName: String,
    val age: Int,
    val position: Position,
    val currentCa: Int,
    val potentialPa: Int,
    val marketValue: Int,
    val wage: Int,
    val attributes: Map<String, Int>,
    val contractUntil: String?
)
```

## 五、观察名单

```kotlin
// watchlist/WatchlistService.kt
class WatchlistService(
    private val saveDb: SaveDatabase,
    private val economyService: EconomyService
) {

    /**
     * V0.1 09 §二.2 观察名单
     * 显示：基本信息/报告等级/预计身价/签约难度/竞争球队/球探建议
     */
    suspend fun addToWatchlist(saveId: Int, playerId: Int): Boolean {
        if (isOnWatchlist(saveId, playerId)) return false

        val player = saveDb.savePlayerStateDao().getByIdWithAttributes(saveId, playerId)
        val marketValue = economyService.estimateMarketValue(player)
        val signingDifficulty = calculateSigningDifficulty(player, saveId)
        val competitors = findCompetitorClubs(player, saveId)

        saveDb.watchlistDao().insert(WatchlistEntry(
            entryId = 0,
            saveId = saveId,
            playerId = playerId,
            addedDate = LocalDate.now().toString(),
            reportLevel = getScoutingReportLevel(playerId, saveId),
            estimatedValue = marketValue,
            signingDifficulty = signingDifficulty,
            competitorClubs = competitors,
            scoutRecommendation = generateScoutRecommendation(player, marketValue)
        ))
        return true
    }

    suspend fun removeFromWatchlist(saveId: Int, playerId: Int) {
        saveDb.watchlistDao().delete(saveId, playerId)
    }

    suspend fun getWatchlist(saveId: Int): List<WatchlistEntry> {
        return saveDb.watchlistDao().getBySave(saveId)
    }

    private fun calculateSigningDifficulty(
        player: SavePlayerStateWithAttributes,
        saveId: Int
    ): SigningDifficulty {
        val marketValue = economyService.estimateMarketValue(player)
        return when {
            player.currentClubId == null -> SigningDifficulty.EASY  // 自由球员
            marketValue > 50_000_000 -> SigningDifficulty.VERY_HARD
            marketValue > 20_000_000 -> SigningDifficulty.HARD
            marketValue > 5_000_000 -> SigningDifficulty.NORMAL
            else -> SigningDifficulty.EASY
        }
    }

    private fun findCompetitorClubs(
        player: SavePlayerStateWithAttributes,
        saveId: Int
    ): List<Int> {
        // 查找也对这名球员感兴趣的 AI 俱乐部
        return saveDb.aiDecisionLogDao()
            .getClubsInterestedInPlayer(saveId, player.playerId)
    }
}
```

```kotlin
// window/TransferWindowManager.kt
class TransferWindowManager(private val config: TransferConfig) {

    /**
     * V0.1 转会窗状态：开窗/关窗/截止日
     */
    fun getStatus(currentDate: LocalDate): TransferWindowStatus {
        val summerWindow = isSummerWindow(currentDate)
        val winterWindow = isWinterWindow(currentDate)

        return when {
            summerWindow -> {
                val endDate = LocalDate.of(currentDate.year, 8, 31)
                val daysToDeadline = java.time.Period.between(currentDate, endDate).days
                if (daysToDeadline <= 3) TransferWindowStatus.CLOSING_SOON
                else TransferWindowStatus.OPEN
            }
            winterWindow -> {
                val endDate = LocalDate.of(currentDate.year, 1, 31)
                val daysToDeadline = java.time.Period.between(currentDate, endDate).days
                if (daysToDeadline <= 2) TransferWindowStatus.CLOSING_SOON
                else TransferWindowStatus.OPEN
            }
            else -> TransferWindowStatus.CLOSED
        }
    }

    private fun isSummerWindow(date: LocalDate): Boolean {
        val md = date.monthValue * 100 + date.dayOfMonth
        return md in 701..831  // 7月1日-8月31日
    }

    private fun isWinterWindow(date: LocalDate): Boolean {
        val md = date.monthValue * 100 + date.dayOfMonth
        return md in 101..131  // 1月1日-1月31日
    }
}

enum class TransferWindowStatus { OPEN, CLOSING_SOON, CLOSED }
```
