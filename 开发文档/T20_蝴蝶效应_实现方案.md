> 任务编号：T20
> 周期：W89-W93（5 周）
> 优先级：P1（差异化卖点，依赖 Gate 3 通过 + T15 + T18）

## 一、整体架构

### 1. 蝴蝶效应流水线

```
触发检测器（TriggerDetector）
    │  3 类触发条件
    ▼
事件创建（ButterflyEventCreator）
    │  写入 butterfly_event 表
    ▼
影响生成器（ImpactGenerator）
    │  衰减公式 0.55^depth
    │  5 种影响类型
    ▼

    │  搜索候选 + AI 评分
    ▼

    │  5 级回退
    ▼

    │  重大节点提示
    ▼
```

```kotlin
// trigger/TriggerDetector.kt
class TriggerDetector(
    private val transferBreakDetector: TransferBreakDetector,
    private val prospectSignedDetector: ProspectSignedDetector,
    private val managerLeaveDetector: ManagerLeaveDetector,
    private val butterflyService: ButterflyEventService
) {

    /**
     * V0.2 §十："第一版只做三类蝴蝶"
     * 1. 关键转会被打断
     * 2. 历史新星提前签约
     * 3. 豪门关键主帅提前离任
     *
     * 由 T07 每日推进调用
     */
    suspend fun detectTriggers(ctx: AdvanceContext): List<ButterflyEvent> {
        val events = mutableListOf<ButterflyEvent>()

        // 1. 关键转会被打断检测
        events += transferBreakDetector.detect(ctx)

        // 2. 历史新星提前签约检测
        events += prospectSignedDetector.detect(ctx)

        // 3. 豪门主帅提前离任检测
        events += managerLeaveDetector.detect(ctx)

        return events
    }
}

// trigger/TransferBreakDetector.kt
class TransferBreakDetector(
    private val historyDb: HistoryDatabase,
    private val saveDb: SaveDatabase,
    private val config: ButterflyConfig
) {

    /**
     * 检测关键历史转会被打断
     * 触发条件：
     * - 历史转会窗口已过
     * - 历史转会未发生（球员不在预期俱乐部）
     * - 该球员未被其他蝴蝶事件标记
     */
    suspend fun detect(ctx: AdvanceContext): List<ButterflyEvent> {
        val events = mutableListOf<ButterflyEvent>()

        // 查询历史关键转会（应在触发日期附近发生）
        val historicalTransfers = historyDb.transferHistoryDao()
            .getKeyTransfersAroundDate(ctx.currentDate, ctx.currentSeasonId)

        for (transfer in historicalTransfers) {
            // 检查是否已被标记为"被打断"
            val existing = saveDb.butterflyEventDao()
                .getBySourcePlayer(transfer.playerId, ctx.saveId.toString())
            if (existing != null) continue

            // 检查球员当前俱乐部是否为预期俱乐部
            val currentPlayerClub = saveDb.savePlayerStateDao()
                .getPlayerCurrentClub(ctx.saveId, transfer.playerId)

            if (currentPlayerClub != transfer.expectedClubId) {
                // 检查是否是关键转会（importance >= 阈值）
                val importance = calculateImportance(transfer)
                if (importance >= config.minTriggerImportance) {  // 默认 70
                    val event = ButterflyEvent(
                        eventId = UUID.randomUUID().toString(),
                        saveId = ctx.saveId.toString(),
                        triggerType = ButterflyTriggerType.HISTORICAL_TRANSFER_BROKEN,
                        sourcePlayerId = transfer.playerId,
                        sourceClubId = transfer.fromClubId,
                        expectedClubId = transfer.expectedClubId,
                        triggerDate = ctx.currentDate,
                        importance = importance,
                        impactBudget = calculateImpactBudget(importance),
                        maxDepth = 3,
                        status = ButterflyEventStatus.PENDING,
                        summary = "球员 ${transfer.playerName} 未按历史转会至 ${transfer.expectedClubName}"
                    )
                    events.add(event)
                    saveDb.butterflyEventDao().insert(event.toEntity())
                }
            }
        }

        return events
    }

    /**
     * 计算事件重要性（V0.2 §三 importance 字段）
     * 基于球员声望、转会费、俱乐部级别
     */
    private fun calculateImportance(transfer: TransferHistoryEntity): Int {
        val playerReputation = historyDb.playerDao().getReputation(transfer.playerId)
        val transferFeeWeight = minOf(transfer.fee / 50_000_000.0, 1.0) * 30  // 5000万满分 30
        val clubLevelWeight = if (transfer.expectedClubReputation > 80) 30 else 15
        return (playerReputation * 0.4 + transferFeeWeight + clubLevelWeight).toInt().coerceIn(0, 100)
    }

    private fun calculateImpactBudget(importance: Int): Int {
        // 重要度越高，预算越大
        return (importance * 1.0).toInt().coerceIn(20, 100)
    }
}

// trigger/ProspectSignedDetector.kt
class ProspectSignedDetector(
    private val historyDb: HistoryDatabase,
    private val saveDb: SaveDatabase
) {

    /**
     * 检测历史新星被提前签约
     * V0.2 §二.1："历史新星被提前签下"
     */
    suspend fun detect(ctx: AdvanceContext): List<ButterflyEvent> {
        val events = mutableListOf<ButterflyEvent>()

        // 查询本应在此日期之后才出现的新星
        val earlySignedProspects = historyDb.prospectDao()
            .getProspectsSignedBeforeDiscoverableDate(ctx.currentDate, ctx.saveId)

        for (prospect in earlySignedProspects) {
            val existing = saveDb.butterflyEventDao()
                .getBySourcePlayer(prospect.playerId, ctx.saveId.toString())
            if (existing != null) continue

            val importance = calculateProspectImportance(prospect)
            if (importance >= 60) {
                val event = ButterflyEvent(
                    eventId = UUID.randomUUID().toString(),
                    saveId = ctx.saveId.toString(),
                    triggerType = ButterflyTriggerType.PROSPECT_SIGNED_EARLY,
                    sourcePlayerId = prospect.playerId,
                    sourceClubId = prospect.defaultFirstTeamClubId,
                    expectedClubId = prospect.defaultFirstTeamClubId,
                    triggerDate = ctx.currentDate,
                    importance = importance,
                    impactBudget = calculateImpactBudget(importance),
                    summary = "历史新星 ${prospect.playerName} 被提前签约"
                )
                events.add(event)
                saveDb.butterflyEventDao().insert(event.toEntity())
            }
        }

        return events
    }
}

// trigger/ManagerLeaveDetector.kt
class ManagerLeaveDetector(
    private val historyDb: HistoryDatabase,
    private val saveDb: SaveDatabase
) {

    /**
     * 检测豪门主帅提前离任
     * V0.2 §二.1："豪门主帅提前离任"
     */
    suspend fun detect(ctx: AdvanceContext): List<ButterflyEvent> {
        val events = mutableListOf<ButterflyEvent>()

        val historicalManagerChanges = historyDb.historicalEventDao()
            .getManagerChangesAroundDate(ctx.currentDate)

        for (change in historicalManagerChanges) {
            // 检查主帅是否提前离任（比历史早）
            if (change.actualLeaveDate.isBefore(change.historicalLeaveDate.minusMonths(3))) {
                val club = saveDb.saveClubStateDao().getByClub(ctx.saveId, change.clubId)
                if (club != null && club.reputation > 75) {  // 仅豪门触发
                    val importance = 70 + (club.reputation - 75)
                    val event = ButterflyEvent(
                        eventId = UUID.randomUUID().toString(),
                        saveId = ctx.saveId.toString(),
                        triggerType = ButterflyTriggerType.MANAGER_LEAVE_EARLY,
                        sourcePlayerId = null,
                        sourceClubId = change.clubId,
                        expectedClubId = null,
                        triggerDate = ctx.currentDate,
                        importance = importance,
                        impactBudget = calculateImpactBudget(importance),
                        summary = "豪门 ${change.clubName} 主帅提前离任"
                    )
                    events.add(event)
                    saveDb.butterflyEventDao().insert(event.toEntity())
                }
            }
        }

        return events
    }
}
```

## 四、影响生成器（V0.2 §四 + §五）

### 1. 衰减计算

```kotlin
// impact/ImpactDecayCalculator.kt
class ImpactDecayCalculator(private val config: ButterflyConfig) {

    /**
     * V0.2 §五 影响衰减公式
     * impact_strength(depth) = base_importance * decay_rate ^ depth
     * 默认 decay_rate = 0.55
     *
     * 例：
     * - 深度 0：100
     * - 深度 1：55
     * - 深度 2：30
     * - 深度 3：16
     *
     * 低于 15 的影响不生成具体事件，只记录为趋势修正
     */
    fun calculateStrength(baseImportance: Int, depth: Int): Double {
        val strength = baseImportance * Math.pow(config.decayRate, depth.toDouble())
        return strength
    }

    fun shouldGenerateImpact(strength: Double): Boolean {
        return strength >= config.minImpactThreshold  // 默认 15
    }

    fun isDepthExceeded(depth: Int): Boolean {
        return depth > config.maxDepth  // 默认 3
    }
}
```

### 2. 影响生成主逻辑

```kotlin
// impact/ImpactGenerator.kt
class ImpactGenerator(
    private val decayCalculator: ImpactDecayCalculator,
    private val transferReplacementHandler: TransferReplacementHandler,
    private val financialImpactHandler: FinancialImpactHandler,
    private val careerPathShiftHandler: CareerPathShiftHandler,
    private val clubStrategyShiftHandler: ClubStrategyShiftHandler,
    private val nationalTeamShiftHandler: NationalTeamShiftHandler,
    private val budgetManager: ImpactBudgetManager,
    private val saveDb: SaveDatabase
) {

    /**
     * V0.2 §四 生成影响
     * 递归生成 3 层影响
     */
    suspend fun generateImpacts(event: ButterflyEvent): List<ButterflyImpactNode> {
        val allNodes = mutableListOf<ButterflyImpactNode>()
        generateImpactsRecursive(event, depth = 0, allNodes)
        return allNodes
    }

    private suspend fun generateImpactsRecursive(
        event: ButterflyEvent,
        depth: Int,
        collected: MutableList<ButterflyImpactNode>
    ) {
        // 1. 检查深度限制
        if (decayCalculator.isDepthExceeded(depth)) return

        // 2. 计算当前层影响强度
        val strength = decayCalculator.calculateStrength(event.importance, depth)
        if (!decayCalculator.shouldGenerateImpact(strength)) return

        // 3. 检查预算
        if (!budgetManager.hasBudget(event.saveId, 1)) return

        // 4. 确定影响类型（按 depth 分层）
        val impactTypes = determineImpactTypes(event, depth)

        // 5. 生成影响节点
        for (impactType in impactTypes) {
            val node = ButterflyImpactNode(
                nodeId = UUID.randomUUID().toString(),
                eventId = event.eventId,
                depth = depth,
                impactType = impactType,
                targetClubId = determineTargetClub(event, impactType, depth),
                targetPlayerId = determineTargetPlayer(event, impactType, depth),
                impactStrength = strength
            )

            // 执行具体影响
            val result = executeImpact(event, node)
            val processedNode = node.copy(
                status = result.status,
                resultSummary = result.summary
            )

            collected.add(processedNode)
            saveDb.butterflyImpactNodeDao().insert(processedNode.toEntity())

            // 消耗预算
            budgetManager.consumeBudget(event.saveId, 1)

            // 递归生成下一层
            if (result.status == ImpactNodeStatus.SUCCESS) {
                generateImpactsRecursive(event, depth + 1, collected)
            }
        }
    }

    /**
     * V0.2 §四 影响类型分层
     * 第一层：直接受影响俱乐部/球员
     * 第二层：替代转会对象、竞争俱乐部
     * 第三层：长期声望、国家队、奖项微调
     */
    private fun determineImpactTypes(event: ButterflyEvent, depth: Int): List<ButterflyImpactType> {
        return when (depth) {
            0 -> when (event.triggerType) {
                ButterflyTriggerType.HISTORICAL_TRANSFER_BROKEN ->
                    listOf(ButterflyImpactType.TRANSFER_REPLACEMENT, ButterflyImpactType.FINANCIAL_IMPACT)
                ButterflyTriggerType.PROSPECT_SIGNED_EARLY ->
                    listOf(ButterflyImpactType.CAREER_PATH_SHIFT, ButterflyImpactType.FINANCIAL_IMPACT)
                ButterflyTriggerType.MANAGER_LEAVE_EARLY ->
                    listOf(ButterflyImpactType.CLUB_STRATEGY_SHIFT)
            }
            1 -> listOf(ButterflyImpactType.TRANSFER_REPLACEMENT, ButterflyImpactType.CLUB_STRATEGY_SHIFT)
            2 -> listOf(ButterflyImpactType.NATIONAL_TEAM_SHIFT, ButterflyImpactType.CAREER_PATH_SHIFT)
            else -> listOf(ButterflyImpactType.NATIONAL_TEAM_SHIFT)
        }
    }

    private suspend fun executeImpact(
        event: ButterflyEvent,
        node: ButterflyImpactNode
    ): ImpactExecutionResult {
        return when (node.impactType) {
            ButterflyImpactType.TRANSFER_REPLACEMENT ->
                transferReplacementHandler.handle(event, node)
            ButterflyImpactType.FINANCIAL_IMPACT ->
                financialImpactHandler.handle(event, node)
            ButterflyImpactType.CAREER_PATH_SHIFT ->
                careerPathShiftHandler.handle(event, node)
            ButterflyImpactType.CLUB_STRATEGY_SHIFT ->
                clubStrategyShiftHandler.handle(event, node)
            ButterflyImpactType.NATIONAL_TEAM_SHIFT ->
                nationalTeamShiftHandler.handle(event, node)
        }
    }
}

data class ImpactExecutionResult(
    val status: ImpactNodeStatus,
    val summary: String
)
```

### 3. 替代转会处理器（V0.2 §四.1 + §六）

```kotlin
// impact/TransferReplacementHandler.kt
class TransferReplacementHandler(
    private val replacementPoolResolver: ReplacementPoolResolver,
    private val fallbackHandler: FallbackHandler,
    private val aiSimulator: AiSimulator,             // T18
    private val saveDb: SaveDatabase
) {

    /**
     * V0.2 §四.1 替代转会
     * 规则：
     * - 从同位置、同年龄层或即战力候选池中搜索
     * - 按 AI 俱乐部决策模型选择
     * - 不强制指定唯一结果
     * - 允许"未找到合适目标"
     */
    suspend fun handle(event: ButterflyEvent, node: ButterflyImpactNode): ImpactExecutionResult {
        val affectedClubId = node.targetClubId ?: event.expectedClubId
            ?: return ImpactExecutionResult(ImpactNodeStatus.NO_REPLACEMENT, "无受影响俱乐部")

        // 1. 加载替代决策池规则
        val rule = replacementPoolResolver.resolveRule(event, affectedClubId)
            ?: return ImpactExecutionResult(ImpactNodeStatus.NO_REPLACEMENT, "无替代规则")

        // 2. 按回退优先级尝试
        for (action in rule.fallbackActions) {
            val result = tryFallbackAction(action, event, node, rule, affectedClubId)
            if (result.status == ImpactNodeStatus.SUCCESS) {
                return result
            }
        }

        // 3. 全部失败，记录未产生替代
        return ImpactExecutionResult(
            ImpactNodeStatus.NO_REPLACEMENT,
            "V0.2 §七：历史分歧未产生重大替代"
        )
    }

    private suspend fun tryFallbackAction(
        action: FallbackAction,
        event: ButterflyEvent,
        node: ButterflyImpactNode,
        rule: ReplacementPoolRule,
        clubId: Int
    ): ImpactExecutionResult {
        return when (action) {
            FallbackAction.SEARCH_FULL_STANDARD ->
                searchReplacement(rule, clubId, node.impactStrength)
            FallbackAction.LOWER_STANDARD_AND_SEARCH ->
                searchReplacement(rule.copy(need = rule.need.copy(caMin = rule.need.caMin - 10)), clubId, node.impactStrength)
            FallbackAction.LOAN_PLAYER ->
                fallbackHandler.loanPlayer(rule, clubId)
            FallbackAction.PROMOTE_YOUTH ->
                fallbackHandler.promoteYouth(rule, clubId)
            FallbackAction.MAINTAIN_CURRENT_SQUAD ->
                ImpactExecutionResult(ImpactNodeStatus.SUCCESS, "维持现有阵容")
            FallbackAction.RECORD_NO_REPLACEMENT ->
                ImpactExecutionResult(ImpactNodeStatus.NO_REPLACEMENT, "记录未产生替代")
        }
    }

    /**
     * 搜索替代球员
     * V0.2 §四.1："按 AI 俱乐部决策模型选择"
     */
    private suspend fun searchReplacement(
        rule: ReplacementPoolRule,
        clubId: Int,
        impactStrength: Double
    ): ImpactExecutionResult {
        // 1. 搜索候选（按位置/年龄/能力/联赛范围）
        val candidates = replacementPoolResolver.searchCandidates(rule.need)
        if (candidates.isEmpty()) {
            return ImpactExecutionResult(ImpactNodeStatus.NO_REPLACEMENT, "无候选球员")
        }

        // 2. 调用 T18 AI 评分
        val clubProfile = saveDb.clubAiProfileDao().getByClub(clubId) ?: return ImpactExecutionResult(ImpactNodeStatus.NO_REPLACEMENT, "无俱乐部画像")
        val scoredTargets = aiSimulator.scoreTransferTargets(candidates, clubProfile, rule.need)

        // 3. 选择最高分且预算可承受
        val affordable = scoredTargets.filter { it.isAffordable }
        val target = affordable.firstOrNull()
            ?: return ImpactExecutionResult(ImpactNodeStatus.NO_REPLACEMENT, "无 affordable 候选")
```

```kotlin
     */
    fun generateDivergenceLogs(event: ButterflyEvent, nodes: List<ButterflyImpactNode>): List<DivergenceLog> {
        val logs = mutableListOf<DiverflyLog>()

        // 1. 主事件提示
        logs.add(DivergenceLog(
            logId = UUID.randomUUID().toString(),
            saveId = event.saveId,
            eventId = event.eventId,
            nodeId = null,
            title = templateLoader.getEventTitle(event.triggerType),
            content = templateLoader.renderEventContent(event),
            date = event.triggerDate,
            importance = event.importance,
            isRead = false
        ))

        // 2. 仅展示深度 0 和 1 的 SUCCESS 节点
        for (node in nodes.filter { it.depth <= 1 && it.status == ImpactNodeStatus.SUCCESS }) {
            logs.add(DivergenceLog(
                logId = UUID.randomUUID().toString(),
                saveId = event.saveId,
                eventId = event.eventId,
                nodeId = node.nodeId,
                title = templateLoader.getImpactTitle(node.impactType),
                content = templateLoader.renderImpactContent(event, node),
                date = event.triggerDate,
                importance = node.impactStrength.toInt(),
                isRead = false
            ))
        }

        // 3. 深度 2+ 的影响不单独展示，合并到趋势修正
        return logs
    }
```

```kotlin
// api/ButterflyEventService.kt
class ButterflyEventService(
    private val triggerDetector: TriggerDetector,
    private val impactGenerator: ImpactGenerator,
    private val budgetManager: ImpactBudgetManager,
    private val divergencePresenter: DivergencePresenter,
    private val butterflyLogger: ButterflyLogger,
    private val saveDb: SaveDatabase
) {

    /**
     * 由 T07 每日推进调用
     */
    suspend fun checkAndProcessTriggers(ctx: AdvanceContext): List<ButterflyResult> {
        val results = mutableListOf<ButterflyResult>()

        // 1. 检测触发
        val triggeredEvents = triggerDetector.detectTriggers(ctx)

        for (event in triggeredEvents) {
            // 2. 预算检查
            if (!budgetManager.canCreateEvent(ctx.saveId.toString())) {
                butterflyLogger.log("预算已满，跳过事件 ${event.eventId}")
                continue
            }

            // 3. 生成影响
            val impactNodes = impactGenerator.generateImpacts(event)

            // 4. 生成玩家可见反馈
            val divergences = divergencePresenter.generateDivergenceLogs(event, impactNodes)

            // 5. 保存分歧日志
            divergences.forEach { saveDb.divergenceLogDao().insert(it.toEntity()) }

            // 6. 标记事件完成
            saveDb.butterflyEventDao().updateStatus(event.eventId, ButterflyEventStatus.COMPLETED.name)

            // 7. 记录日志
            butterflyLogger.logEvent(event, impactNodes)

            results.add(ButterflyResult(
                event = event,
                impactNodes = impactNodes,
                divergences = divergences,
                budgetUsed = impactNodes.size,
                budgetRemaining = config.maxNodesPerSeason - budgetManager.getSeasonUsage(ctx.saveId.toString())
            ))
        }

        return results
    }

    /**
     * 检查某历史事件是否已被蝴蝶改写
     * 供 T07 HistoryEventTask 调用
     */
    fun isEventRewritten(historicalEventId: String, saveId: String): Boolean {
        return saveDb.butterflyEventDao()
            .getByHistoricalEvent(historicalEventId, saveId) != null
    }
}
```

## 十二、关键风险与应对

### 1. 蝴蝶级联崩塌

- **表现**：一次转会打断触发无数连锁反应
- **应对**：
  - 严格 3 层深度限制
  - 每赛季 200 节点预算
  - 衰减 0.55^depth，第 3 层仅 16 强度
- **红线**：低于 15 强度不生成事件

### 2. 替代转会荒谬

- **表现**：豪门为替代 C 罗签下低级别联赛球员
- **应对**：
  - 替代候选必须满足 ca_min/pa_min/league_scope
  - 调用 T18 AI 评分确保合理性
  - 5 级回退，允许"未找到替代"
- **铁律**：V0.2 §七"不允许为了蝴蝶效应强行制造荒谬转会"

### 3. 性能问题

- **表现**：蝴蝶检测拖慢每日推进
- **应对**：
  - 检测只在触发日期附近执行（不是每日全量扫描）
  - 影响生成异步化
  - 候选搜索限制 20 个

### 4. 玩家看不懂

- **表现**：分歧提示过于技术化
- **应对**：
  - 模板化文案，贴近自然语言
  - 只展示深度 0/1 的重大节点
  - 提供"历史分歧时间线"页面供查看

## 十四、V1 范围明确（不做的）

**V1 做**：
- 3 类触发（关键转会被打断/历史新星提前签约/豪门主帅提前离任）
- 5 种影响类型（替代转会/财政/成长路径/俱乐部策略/国家队趋势）
- 衰减公式 + 3 层扩散
- 影响预算管理（20 事件/200 节点）
- 5 级回退机制
- 玩家可见反馈（深度 0/1）
- 分歧日志持久化

**V1 不做**（推迟）：
- 国家队实际改变（V1 仅记录趋势）
- 重大资本事件被改变
- 欧冠/世界杯关键结果改变
- 蝴蝶事件的连锁可视化（T21 做）
- 玩家手动触发蝴蝶
- 蝴蝶事件撤销
