## 三、时代通胀系数（V0.2 §二）

```kotlin
// index/EconomyIndexProvider.kt
class EconomyIndexProvider(
    private val table: EconomyIndexTable,
    private val futureProjector: FutureGrowthProjector,
    private val config: EconomyConfig
) {

    /**
     * V0.2 §二 时代通胀系数
     * 以 2002 年为 1.00
     * 2030 后进入架空增长，默认年增长 2%-4%
     */
    fun getIndex(year: Int): Double {
        return when {
            year <= 2030 -> table.getFixedIndex(year)
            else -> futureProjector.project(2030, year)
        }
    }

    fun getIndexForDate(date: LocalDate): Double = getIndex(date.year)
}

// index/EconomyIndexTable.kt
class EconomyIndexTable {

    /**
     * V0.2 §二 固定系数表
     */
    private val fixedTable = mapOf(
        1992 to 0.35,
        1995 to 0.45,
        1998 to 0.65,
        2002 to 1.00,
        2006 to 1.25,
        2010 to 1.60,
        2014 to 2.20,
        2017 to 3.20,
        2020 to 3.60,
        2024 to 4.20,
        2030 to 5.00
    )

    fun getFixedIndex(year: Int): Double {
        // 线性插值
        val keys = fixedTable.keys.sorted()
        if (year <= keys.first()) return fixedTable[keys.first()]!!
        if (year >= keys.last()) return fixedTable[keys.last()]!!

        val lower = keys.last { it <= year }
        val upper = keys.first { it >= year }
        if (lower == upper) return fixedTable[lower]!!

        val lowerValue = fixedTable[lower]!!
        val upperValue = fixedTable[upper]!!
        val ratio = (year - lower).toDouble() / (upper - lower)
        return lowerValue + (upperValue - lowerValue) * ratio
    }
}

// index/FutureGrowthProjector.kt
class FutureGrowthProjector(private val config: EconomyConfig) {

    /**
     * V0.2 §二："2030 后进入架空增长，默认年增长 2%-4%，可受全球经济事件影响"
     */
    fun project(fromYear: Int, toYear: Int): Double {
        val baseIndex = 5.00  // 2030 年 = 5.00
        var index = baseIndex

        for (year in (fromYear + 1)..toYear) {
            val growthRate = config.futureGrowthBaseRate + getRandomEventAdjustment(year)  // 默认 0.03
            index *= (1.0 + growthRate)
        }

        return index
    }

    private fun getRandomEventAdjustment(year: Int): Double {
        // V1 简化：固定 3% 增长，未来可加入全球经济事件
        return 0.0
    }
}
```

```kotlin
// league/LeagueEconomyProvider.kt
class LeagueEconomyProvider(
    private val table: LeagueEconomyTable,
    private val economyIndexProvider: EconomyIndexProvider
) {

    /**
     * V0.2 §三 联赛商业系数
     * 2002 基础系数 + 随年代增长曲线
     */
    fun getMultiplier(leagueId: String, year: Int): Double {
        val baseMultiplier = table.getBaseMultiplier(leagueId)
        val growthCurve = table.getGrowthCurve(leagueId)
        val economyIndex = economyIndexProvider.getIndex(year)

        // 联赛系数 = 基础系数 × 增长曲线(年) × 经济指数修正
        return baseMultiplier * growthCurve(year) * (economyIndex / economyIndexProvider.getIndex(2002))
    }
}

class LeagueEconomyTable {

        "Ligue1" to LeagueEconomyData(0.70, GrowthType.CAPITAL_EVENT, "后续资本事件可提升"),
        "Eredivisie" to LeagueEconomyData(0.45, GrowthType.EXPORT, "青训出口"),
        "PrimeiraLiga" to LeagueEconomyData(0.40, GrowthType.SELLING, "黑店体系"),
        "Brasileirao" to LeagueEconomyData(0.35, GrowthType.TALENT_EXPORT, "输出天才"),
        "Argentino" to LeagueEconomyData(0.30, GrowthType.TALENT_EXPORT, "输出天才")
    )

    fun getBaseMultiplier(leagueId: String): Double =
        leagueData[leagueId]?.baseMultiplier ?: 0.50

    fun getGrowthCurve(leagueId: String): (Int) -> Double {
        val data = leagueData[leagueId] ?: return { 1.0 }
        return when (data.growthType) {
            GrowthType.FAST -> { year -> 1.0 + (year - 2002) * 0.02 }              // 年增 2%
            GrowthType.TOP_HEAVY -> { year -> 1.0 + (year - 2002) * 0.015 }
            GrowthType.DECLINING -> { year ->
                if (year < 2005) 1.05
                else 1.05 - (year - 2005) * 0.01  // 2005 后下滑
            }
            GrowthType.STABLE -> { year -> 1.0 + (year - 2002) * 0.01 }
            GrowthType.CAPITAL_EVENT -> { year ->
                if (year < 2011) 1.0
                else 1.0 + (year - 2011) * 0.03  // 2011 后资本入主
            }
            GrowthType.EXPORT -> { year -> 1.0 + (year - 2002) * 0.005 }
            GrowthType.SELLING -> { year -> 1.0 + (year - 2002) * 0.005 }
            GrowthType.TALENT_EXPORT -> { year -> 1.0 }
        }
    }
}
```

```kotlin
// club/ClubFinancialPowerCalculator.kt
class ClubFinancialPowerCalculator(
    private val leagueEconomyProvider: LeagueEconomyProvider,
    private val config: EconomyConfig
) {

    /**
     * V0.2 §四 俱乐部财力计算
     * club_financial_power =
     *   club_reputation * 0.25
     * + league_economy * 0.25
     * + stadium_income * 0.15
     * + commercial_income * 0.15
     * + owner_investment * 0.15
     * + recent_success * 0.05
     */
    fun calculate(club: ClubEntity, financial: ClubFinancialState, year: Int): Int {
        val clubReputation = club.reputation.toDouble() / 100.0  // 归一化 0-1
        val leagueEconomy = leagueEconomyProvider.getMultiplier(club.country, year) / 2.0  // 归一化
        val stadiumIncome = normalizeIncome(financial.stadiumIncome)
        val commercialIncome = normalizeIncome(financial.commercialIncome)
        val ownerInvestment = normalizeIncome(financial.ownerInvestment)
        val recentSuccess = financial.recentSuccess  // 已是 0-1

        val w = config.financialPowerWeights
        val power = (
            clubReputation * w.reputation +
            leagueEconomy * w.leagueEconomy +
            stadiumIncome * w.stadiumIncome +
            commercialIncome * w.commercialIncome +
            ownerInvestment * w.ownerInvestment +
            recentSuccess * w.recentSuccess
        ) * 100

        return power.toInt().coerceIn(0, 100)
    }

    private fun normalizeIncome(income: Int): Double {
        // 5000万为满分 1.0
        return (income / 50_000_000.0).coerceIn(0.0, 1.0)
    }
}

data class FinancialPowerWeights(
    val reputation: Double,         // 0.25
    val leagueEconomy: Double,      // 0.25
    val stadiumIncome: Double,      // 0.15
    val commercialIncome: Double,   // 0.15
    val ownerInvestment: Double,    // 0.15
    val recentSuccess: Double       // 0.05
)
```

## 六、球员身价（V0.2 §五-八）

```kotlin
// player/PlayerValueCalculator.kt
class PlayerValueCalculator(
    private val abilityValueCurve: AbilityValueCurve,
    private val ageMultiplier: AgeMultiplier,
    private val contractMultiplier: ContractRemainMultiplier,
    private val performanceMultiplier: PerformanceMultiplier,
    private val economyIndexProvider: EconomyIndexProvider,
    private val leagueEconomyProvider: LeagueEconomyProvider,
    private val config: EconomyConfig
) {

    /**
     * V0.2 §五 球员基础身价
     * base_value =
     *   ability_value_curve(CA)
     * * potential_multiplier(PA, age)
     * * age_multiplier(age)
     * * position_multiplier(position)
     * * reputation_multiplier(reputation)
     * * league_visibility_multiplier(league)
     * * economy_index(year)
     */
    fun calculateBaseValue(
        player: SavePlayerStateWithAttributes,
        clubLeagueId: String,
        year: Int
    ): PlayerValuation {
        val ca = player.currentCa
        val pa = player.potentialPa
        val age = calculateAge(player.birthDate)

        // 1. CA 价值曲线（V0.2 §五 CA 价值曲线）
        val abilityValue = abilityValueCurve.calculate(ca)

        // 2. 潜力乘数
        val potentialMultiplier = calculatePotentialMultiplier(pa, ca, age)

        // 3. 年龄系数（V0.2 §六）
        val ageMult = ageMultiplier.calculate(age, pa)

        // 4. 位置系数
        val positionMult = config.positionMultiplier[player.primaryPosition] ?: 1.0

        // 5. 声望系数
        val reputationMult = calculateReputationMultiplier(player.reputation)

        // 6. 联赛可见度
        val leagueVisibility = leagueEconomyProvider.getMultiplier(clubLeagueId, year)

        // 7. 时代系数
        val economyIndex = economyIndexProvider.getIndex(year)

        // 8. 合同剩余影响（V0.2 §七）
        val contractMult = contractMultiplier.calculate(player.contractUntil, LocalDate.of(year, 1, 1))

        // 基础身价
        val baseValue = (abilityValue *
            potentialMultiplier *
            ageMult *
            positionMult *
            reputationMult *
            leagueVisibility *
            economyIndex *
            contractMult).toInt()

        // 9. 表现修正（V0.2 §八）
        val perfMult = performanceMultiplier.calculate(player)
        val currentValue = (baseValue * perfMult).toInt()

        // 10. 工资
        val expectedWage = calculateWage(player, clubLeagueId, year)

        return PlayerValuation(
            playerId = player.playerId,
            baseValue = baseValue,
            currentValue = currentValue,
            expectedWage = expectedWage,
            breakdown = ValuationBreakdown(
                abilityValue, potentialMultiplier, ageMult, positionMult,
                reputationMult, leagueVisibility, economyIndex, contractMult, perfMult
            )
        )
    }

    /**
     * V0.2 §五 CA 价值曲线
     * ability_value_curve = base_amount * pow(1.075, CA - 50)
     */
    private fun calculatePotentialMultiplier(pa: Int, ca: Int, age: Int): Double {
        val potentialGap = pa - ca
        // 年轻球员潜力差距价值更高
        return when {
            age <= 21 && potentialGap > 15 -> 1.0 + potentialGap * 0.02  // 最高 +60%
            age <= 24 && potentialGap > 10 -> 1.0 + potentialGap * 0.01
            else -> 1.0
        }
    }

    private fun calculateReputationMultiplier(reputation: Int): Double {
        return 0.5 + reputation / 100.0  // 0.5 - 1.5
    }
}

// player/AbilityValueCurve.kt
class AbilityValueCurve(private val config: EconomyConfig) {

    /**
     * V0.2 §五 CA 价值曲线
     * CA < 60：低价值
     * 60-70：普通职业球员
     * 70-80：主力级
     * 80-88：明星级
     * 88-94：世界级
     * 95+：历史级，指数增长
     *
     * 公式：base_amount * pow(1.075, CA - 50)
     */
    fun calculate(ca: Int): Double {
        val baseAmount = config.abilityCurveBaseAmount  // 默认 100_000
        return baseAmount * Math.pow(1.075, (ca - 50).toDouble())
    }
}

// player/AgeMultiplier.kt
class AgeMultiplier {

    /**
     * V0.2 §六 年龄系数
     * | 16-18 | 0.60-1.20，受潜力影响大 |
     * | 19-22 | 1.10-1.50 |
     * | 23-27 | 1.20-1.60 |
     * | 28-30 | 1.00-1.30 |
     * | 31-33 | 0.65-0.95 |
     * | 34+   | 0.20-0.60 |
     */
    fun calculate(age: Int, potentialPa: Int): Double {
        return when (age) {
            in 16..18 -> {
                // 受潜力影响大
                if (potentialPa > 85) 1.20 else 0.60 + (potentialPa - 50) * 0.012
            }
            in 19..22 -> 1.10 + (potentialPa - 50) * 0.008  // 1.10-1.50
            in 23..27 -> 1.20 + (potentialPa - 50) * 0.008  // 1.20-1.60
            in 28..30 -> 1.00 + (potentialPa - 50) * 0.006  // 1.00-1.30
            in 31..33 -> 0.65 + (potentialPa - 50) * 0.006  // 0.65-0.95
            else -> {
                if (age <= 36) 0.20 + (potentialPa - 50) * 0.008  // 0.20-0.60
                else 0.15
            }
        }.coerceIn(0.10, 1.60)
    }
}

// player/ContractRemainMultiplier.kt
class ContractRemainMultiplier {

    /**
     * V0.2 §七 合同剩余影响
     * | 4 年以上 | 1.20 |
     * | 3 年     | 1.10 |
     * | 2 年     | 1.00 |
     * | 1 年     | 0.70 |
     * | 6 个月   | 0.40 |
     * | 自由球员 | 转会费 0，签字费上升 |
     */
    fun calculate(contractUntil: String?, currentDate: LocalDate): Double {
        if (contractUntil == null) return 0.0  // 自由球员

        val endDate = LocalDate.parse(contractUntil)
        val yearsRemaining = java.time.Period.between(currentDate, endDate).years.toDouble() +
            java.time.Period.between(currentDate, endDate).months.toDouble() / 12.0

        return when {
            yearsRemaining >= 4 -> 1.20
            yearsRemaining >= 3 -> 1.10
            yearsRemaining >= 2 -> 1.00
            yearsRemaining >= 1 -> 0.70
            yearsRemaining >= 0.5 -> 0.40
            else -> 0.0  // 自由球员
        }
    }
}
```

## 七、工资模型（V0.2 §九）

```kotlin
// wage/WageCalculator.kt
class WageCalculator(
    private val economyIndexProvider: EconomyIndexProvider,
    private val leagueEconomyProvider: LeagueEconomyProvider,
    private val squadRoleMultiplier: SquadRoleMultiplier,
    private val config: EconomyConfig
) {

    /**
     * V0.2 §九 工资模型
     * expected_wage =
     *   wage_base_by_CA(CA)
     * * reputation_multiplier
     * * league_wage_multiplier
     * * economy_index
     * * agent_greed_multiplier
     * * squad_role_multiplier
     */
    fun calculate(
        player: SavePlayerStateWithAttributes,
        clubLeagueId: String,
        year: Int
    ): Int {
        val wageBase = calculateWageBaseByCA(player.currentCa)
        val reputationMult = 0.5 + player.reputation / 100.0
        val leagueWageMult = leagueEconomyProvider.getMultiplier(clubLeagueId, year)
        val economyIndex = economyIndexProvider.getIndex(year)
        val agentGreedMult = calculateAgentGreed(player.agentId)
        val squadRoleMult = squadRoleMultiplier.get(player.squadRole)

        val wage = (wageBase *
            reputationMult *
            leagueWageMult *
            economyIndex *
            agentGreedMult *
            squadRoleMult).toInt()

        return wage.coerceAtLeast(config.minWage)  // 最低工资
    }

    /**
     * CA 工资基础表
     */
    private fun calculateWageBaseByCA(ca: Int): Int {
        // 2002 年基准：CA 80 ≈ 5万周薪
        return when {
            ca < 60 -> 2_000
            ca in 60..70 -> 5_000 + (ca - 60) * 1_000
            ca in 71..80 -> 15_000 + (ca - 70) * 4_000
            ca in 81..88 -> 55_000 + (ca - 80) * 10_000
            ca in 89..94 -> 135_000 + (ca - 88) * 20_000
            else -> 255_000 + (ca - 94) * 50_000  // 历史级
        }
    }

    private fun calculateAgentGreed(agentId: String?): Double {
        // V1 简化：1.0-1.15 随机
        return 1.0 + (agentId?.hashCode()?.rem(15)?.let { it / 100.0 } ?: 0.0)
    }
}

// wage/SquadRoleMultiplier.kt
class SquadRoleMultiplier {

    /**
     * V0.2 §九 队内角色修正
     * | 核心球星 | 1.40 |
     * | 主力     | 1.10 |
     * | 轮换     | 0.85 |
     * | 替补     | 0.60 |
     * | 青年球员 | 0.25 |
     */
    fun get(squadRole: String?): Double {
```

```kotlin
// health/FinancialHealthChecker.kt
class FinancialHealthChecker {

    /**
     * V0.2 §十 财政安全线
     * | < 55%      | 健康 |
     * | 55%-70%    | 可接受 |
     * | 70%-85%    | 风险 |
     * | > 85%      | 高危 |
     *
     * AI 俱乐部超过 85% 后：
     * - 降低买人欲望
     * - 提高卖人概率
     * - 优先清理高薪替补
     * - 董事会可能限制预算
     */
    fun check(financial: ClubFinancialState): FinancialHealthReport {
        val ratio = financial.wageToIncomeRatio
        val level = when {
            ratio < 0.55 -> FinancialWarningLevel.HEALTHY
            ratio in 0.55..0.70 -> FinancialWarningLevel.ACCEPTABLE
            ratio in 0.70..0.85 -> FinancialWarningLevel.RISK
            else -> FinancialWarningLevel.HIGH_RISK
        }

        val recommendations = generateRecommendations(level, financial)

        return FinancialHealthReport(
            clubId = financial.clubId,
            wageToIncomeRatio = ratio,
            level = level,
            recommendations = recommendations
        )
    }

    private fun generateRecommendations(
        level: FinancialWarningLevel,
        financial: ClubFinancialState
    ): List<String> {
        return when (level) {
            FinancialWarningLevel.HEALTHY -> emptyList()
            FinancialWarningLevel.ACCEPTABLE -> listOf("关注工资增长趋势")
            FinancialWarningLevel.RISK -> listOf(
                "降低买人欲望",
                "考虑出售高薪替补"
            )
            FinancialWarningLevel.HIGH_RISK -> listOf(
                "停止买人",
                "提高卖人概率",
                "优先清理高薪替补",
                "董事会限制预算"
            )
        }
    }
}

enum class FinancialWarningLevel {
    HEALTHY,         // < 55%
    ACCEPTABLE,      // 55%-70%
    RISK,            // 70%-85%
    HIGH_RISK        // > 85%
}

data class FinancialHealthReport(
    val clubId: Int,
    val wageToIncomeRatio: Double,
    val level: FinancialWarningLevel,
    val recommendations: List<String>
)
```

```kotlin
// event/EconomicEventEngine.kt
class EconomicEventEngine(
    private val capitalInjectionHandler: CapitalInjectionHandler,
    private val newStadiumHandler: NewStadiumHandler,
    private val europeanQualificationHandler: EuropeanQualificationHandler,
    private val relegationHandler: RelegationHandler,
    private val financialCrisisHandler: FinancialCrisisHandler
) {

    /**
     * V0.2 §十一 经济事件
     * - 资本入主：owner_investment 提升
     * - 新球场：短期债务上升，长期收入上升
     * - 欧冠资格：收入提升
     * - 降级：收入下降
     * - 财政危机：预算冻结
     * - 联赛转播上涨：league_economy 增长
     */
    suspend fun processEvents(ctx: EconomyContext, club: ClubFinancialState): ClubFinancialState {
        var state = club

        // 检查资本入主（随机触发，豪门概率高）
        state = capitalInjectionHandler.tryTrigger(ctx, state)

        // 检查新球场（俱乐部声望高 + 资金充足）
        state = newStadiumHandler.tryTrigger(ctx, state)

        // 欧冠资格（赛季末触发）
        state = europeanQualificationHandler.applyIfQualified(ctx, state)

        // 降级（赛季末触发）
        state = relegationHandler.applyIfRelegated(ctx, state)

        // 财政危机（自动检测）
        state = financialCrisisHandler.checkAndApply(ctx, state)

        return state
    }
}

// event/CapitalInjectionHandler.kt
class CapitalInjectionHandler(private val config: EconomyConfig) {

    /**
     * V0.2 §十一 资本入主
     * 提升 owner_investment
     */
    suspend fun tryTrigger(ctx: EconomyContext, state: ClubFinancialState): ClubFinancialState {
        // V1 简化：按历史事件触发（如 2003 切尔西阿布入主）
        // 未来可改为随机 + 俱乐部声望判断
        val historicalEvent = getHistoricalCapitalInjection(ctx, state.clubId)
        if (historicalEvent != null) {
            return state.copy(
                ownerInvestment = (state.ownerInvestment * historicalEvent.investmentMultiplier).toInt(),
                transferBudget = state.transferBudget + historicalEvent.budgetBoost
            )
        }
        return state
    }
}

// event/FinancialCrisisHandler.kt
class FinancialCrisisHandler {

    /**
     * V0.2 §十一 财政危机
     * 预算冻结
     */
    suspend fun checkAndApply(ctx: EconomyContext, state: ClubFinancialState): ClubFinancialState {
        val healthReport = FinancialHealthChecker().check(state)
        if (healthReport.level == FinancialWarningLevel.HIGH_RISK) {
            // 冻结预算
            return state.copy(
                transferBudget = 0,
                wageBudget = (state.wageBudget * 0.8).toInt()  // 削减 20%
            )
        }
        return state
    }
}
```

```kotlin
        val club = saveDb.saveClubStateDao().getByClub(saveId, clubId)!!
        val financial = buildFinancialState(club, saveId)
        return financialHealthChecker.check(financial)
    }

    }
```

## 十三、关键风险与应对

### 1. 价格失控

- **表现**：2030 年转会费 > 3 亿或 < 1 亿
- **应对**：
  - 调 `future_growth_base_rate`（默认 3%）
  - 加严财政安全线到 80%
  - 限制豪门每窗交易数
- **红线**：2030 年顶级转会费必须在 1-3 亿区间

### 2. 俱乐部破产潮

- **表现**：多俱乐部财政危机
- **应对**：
  - AI 工资控制更严格
  - 降级俱乐部自动削减工资 30%
  - 财政危机时强制卖人

### 3. 联赛均质化

- **表现**：英超和葡超财力差距缩小
- **应对**：
  - 保持联赛增长曲线差异
  - 资本事件只触发特定联赛（法甲 2011）

### 4. 身价与工资脱节

- **表现**：高身价球员工资过低，或反之
- **应对**：
  - 身价和工资共用 economy_index
  - 工资 base 表与身价 base 表保持比例
