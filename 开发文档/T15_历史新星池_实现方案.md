## 三、新星池管理

```kotlin
// pool/ProspectPoolManager.kt
class ProspectPoolManager(
    private val historyDb: HistoryDatabase,
    private val saveDb: SaveDatabase,
    private val config: ProspectConfig
) {

    /**
     * V0.1 08 §三 历史新星池
     * 按真实时间进入可发现状态
     */
    suspend fun activateProspects(currentDate: LocalDate, saveId: Int): List<HistoricalProspect> {
        // 查询本应在当前日期前可发现的新星
        val dueProspects = historyDb.prospectDao()
            .getProspectsDiscoverableOn(currentDate)

        val activated = mutableListOf<HistoricalProspect>()
        for (prospect in dueProspects) {
            // 检查是否已被发现/签约
            val isAlreadyActive = saveDb.prospectStateDao()
                .isProspectActive(saveId, prospect.prospectId)
            if (isAlreadyActive) continue

            // 激活新星
            saveDb.prospectStateDao().insert(ProspectStateEntity(
                saveId = saveId,
                prospectId = prospect.prospectId,
                playerId = prospect.playerId,
                status = ProspectStatus.ACTIVE.name,
                activatedDate = currentDate.toString(),
                discoveredByClubId = null,
                currentPath = "default"
            ))

            activated.add(prospect)
        }

        return activated
    }

    /**
     * 获取当前活跃新星列表
     */
    suspend fun getActiveProspects(saveId: Int): List<HistoricalProspect> {
        val activeStates = saveDb.prospectStateDao().getActiveBySave(saveId)
        return activeStates.mapNotNull { state ->
            historyDb.prospectDao().getById(state.prospectId)
        }
    }
}

enum class ProspectStatus {
    PENDING,        // 未到时间
    ACTIVE,         // 已激活，可被发现
    DISCOVERED,     // 已被球探发现
    SIGNED_EARLY,   // 被提前签约
    DEFAULT_PATH,   // 按默认路径
    COMPLETED       // 已完成职业路径
}
```

## 四、发现机制

```kotlin
// discover/ProspectDiscoverer.kt
class ProspectDiscoverer(
    private val saveDb: SaveDatabase,
    private val historyDb: HistoryDatabase,
    private val scoutReportService: ScoutReportService    // T14
) {

    /**
     * V0.1 08 §三 历史新星发现
     * 球探任务触发发现
     */
    suspend fun tryDiscover(
        saveId: Int,
        scoutId: Int,
        regionCode: String,
        currentDate: LocalDate
    ): List<DiscoveryResult> {
        val discoveries = mutableListOf<DiscoveryResult>()

        // 1. 获取该地区活跃且未被发现的新星
        val activeProspects = getActiveUndiscoveredProspects(saveId, regionCode)

        for (prospect in activeProspects) {
            // 2. 计算发现概率
            val scout = saveDb.scoutDao().getById(scoutId)
            val discoverProbability = calculateDiscoveryProbability(scout, prospect, regionCode)

            // 3. 概率性发现
            if (Random.nextDouble() < discoverProbability) {
                // 4. 生成球探报告（初次发现等级）
                val report = scoutReportService.generateInitialReport(
                    saveId, prospect.playerId, scout, currentDate
                )

                // 5. 更新星状态
                saveDb.prospectStateDao().updateStatus(
                    saveId, prospect.prospectId, ProspectStatus.DISCOVERED.name
                )
                saveDb.prospectStateDao().updateDiscoveredBy(
                    saveId, prospect.prospectId, scout.clubId
                )

                discoveries.add(DiscoveryResult(
                    prospectId = prospect.prospectId,
                    playerId = prospect.playerId,
                    playerName = prospect.playerName,
                    report = report,
                    message = "球探在 $regionCode 发现了 ${prospect.playerName}！"
                ))
            }
        }

        return discoveries
    }

    /**
     * V0.1 08 §三.5 发现概率
     * 基于球探能力 + 地区知识 + 任务匹配
     */
    private fun calculateDiscoveryProbability(
        scout: ScoutEntity,
        prospect: HistoricalProspect,
        regionCode: String
    ): Double {
        val regionKnowledge = scout.getRegionKnowledge(regionCode) / 100.0
        val potentialJudgment = scout.potentialJudgment / 100.0
        val network = scout.network / 100.0
        val taskMatch = 1.0  // V1 简化
        val regionTalentDensity = getRegionTalentDensity(regionCode)

        val probability = (
            regionKnowledge * 0.25 +
            potentialJudgment * 0.20 +
            network * 0.15 +
            taskMatch * 0.15 +
            regionTalentDensity * 0.10 +
            Random.nextDouble() * 0.05  // 随机因素
        )

        return probability.coerceIn(0.0, 0.8)  // 单次最高 80%
    }
}

data class DiscoveryResult(
    val prospectId: Int,
    val playerId: Int,
    val playerName: String,
    val report: ScoutReport,
    val message: String
)
```

## 五、默认路径执行

```kotlin
// path/DefaultPathExecutor.kt
class DefaultPathExecutor(
    private val saveDb: SaveDatabase,
    private val historyDb: HistoryDatabase,
    private val transferService: TransferService
) {

    /**
     * V0.1 08 §三 历史新星
     * 未被打断时按历史转会路径
     */
    suspend fun executeDefaultPath(
        prospect: HistoricalProspect,
        currentDate: LocalDate,
        saveId: Int
    ): ProspectResult {
        // 1. 查找当前应执行的转会
        val nextTransfer = prospect.defaultTransferPath.find { transfer ->
            !isTransferExecuted(saveId, prospect.prospectId, transfer)
        } ?: return ProspectResult(
            prospectId = prospect.prospectId,
            action = ProspectAction.DEFAULT_PATH_EXECUTED,
            playerId = prospect.playerId,
            clubId = null,
            butterflyEventTriggered = false,
            message = "${prospect.playerName} 默认路径已全部完成"
        )

        // 2. 检查是否到转会日期
        if (currentDate.isBefore(nextTransfer.transferDate)) {
            return ProspectResult(
                prospectId = prospect.prospectId,
                action = ProspectAction.DEFAULT_PATH_EXECUTED,
                playerId = prospect.playerId,
                clubId = null,
                butterflyEventTriggered = false,
                message = "未到转会日期"
            )
        }

        // 3. 检查是否已被玩家提前签约
        val playerState = saveDb.savePlayerStateDao()
            .getById(saveId, prospect.playerId)
        if (playerState?.currentClubId != nextTransfer.fromClubId &&
            isPlayerSignedByPlayer(saveId, prospect.playerId)) {
            // 已被玩家签走，路径被打断
            return ProspectResult(
                prospectId = prospect.prospectId,
                action = ProspectAction.PATH_INTERRUPTED,
                playerId = prospect.playerId,
                clubId = playerState?.currentClubId,
                butterflyEventTriggered = true,
                message = "${prospect.playerName} 已被提前签约，默认路径打断"
            )
        }

        // 4. 执行默认转会
        transferService.executeHistoricalTransfer(
            saveId, prospect.playerId, nextTransfer, currentDate
        )

        // 5. 标记已执行
        saveDb.prospectTransferHistoryDao().insert(ProspectTransferHistoryEntity(
            saveId = saveId,
            prospectId = prospect.prospectId,
            fromClubId = nextTransfer.fromClubId,
            toClubId = nextTransfer.toClubId,
            transferDate = nextTransfer.transferDate.toString(),
            fee = nextTransfer.transferFee,
            isDefaultPath = true
        ))

        return ProspectResult(
            prospectId = prospect.prospectId,
            action = ProspectAction.DEFAULT_PATH_EXECUTED,
            playerId = prospect.playerId,
            clubId = nextTransfer.toClubId,
            butterflyEventTriggered = false,
            message = "${prospect.playerName} 按历史转会至 ${getClubName(nextTransfer.toClubId)}"
        )
    }
}
```

## 六、提前签约打断

```kotlin
// interrupt/EarlySignInterrupter.kt
class EarlySignInterrupter(
    private val butterflyService: ButterflyEventService,    // T20
    private val saveDb: SaveDatabase
) {

    /**
     * V0.2 06 §二.1 历史新星提前签约
     * 玩家签走后不再按历史转会，触发蝴蝶效应
     */
    suspend fun onPlayerEarlySigned(
        playerId: Int,
        signedByClubId: Int,
        currentDate: LocalDate,
        saveId: Int
    ): ProspectResult {
        // 1. 查找该球员对应的历史新星
        val prospect = historyDb.prospectDao().getByPlayerId(playerId)
            ?: return ProspectResult(
                prospectId = -1,
                action = ProspectAction.EARLY_SIGNED,
                playerId = playerId,
                clubId = signedByClubId,
                butterflyEventTriggered = false,
                message = "非历史新星，无需打断"
            )

        // 2. 检查是否真的"提前"（早于 defaultBreakthroughYear）
        val isEarly = currentDate.year < prospect.defaultBreakthroughYear ||
                      signedByClubId != prospect.defaultFirstTeamClubId

        if (!isEarly) {
            return ProspectResult(
                prospectId = prospect.prospectId,
                action = ProspectAction.EARLY_SIGNED,
                playerId = playerId,
                clubId = signedByClubId,
                butterflyEventTriggered = false,
                message = "按历史签约，无需触发蝴蝶"
            )
        }

        // 3. 更新星状态
        saveDb.prospectStateDao().updateStatus(
            saveId, prospect.prospectId, ProspectStatus.SIGNED_EARLY.name
        )
        saveDb.prospectStateDao().updateCurrentPath(
            saveId, prospect.prospectId, "interrupted"
        )

        // 4. 触发蝴蝶效应（T20）
        val butterflyTriggered = butterflyService.triggerProspectSignedEarly(
            prospect, signedByClubId, currentDate, saveId
        )

        return ProspectResult(
            prospectId = prospect.prospectId,
            action = ProspectAction.EARLY_SIGNED,
            playerId = playerId,
            clubId = signedByClubId,
            butterflyEventTriggered = butterflyTriggered,
            message = "${prospect.playerName} 被提前签约，历史路径已打断，触发蝴蝶效应"
        )
    }
}
```

## 七、AI 竞争签约

```kotlin
// ai/AiCompetitorSigner.kt
class AiCompetitorSigner(
    private val aiTransferEngine: AiTransferEngine,        // T13/T18
    private val saveDb: SaveDatabase,
    private val config: ProspectConfig
) {

    /**
     * V0.2 06 §十一.5 年轻天才不全部流向同一家豪门
     * AI 俱乐部按 target_score 竞争签约历史新星
     */
    suspend fun processAiSigning(
        prospect: HistoricalProspect,
        currentDate: LocalDate,
        saveId: Int
    ): ProspectResult {
        // 1. 检查是否到了新星可签约时间
        if (currentDate.isBefore(prospect.discoverableFrom.plusMonths(6))) {
            return ProspectResult(
                prospectId = prospect.prospectId,
                action = ProspectAction.DEFAULT_PATH_EXECUTED,
                playerId = prospect.playerId,
                clubId = null,
                butterflyEventTriggered = false,
                message = "新星尚未到可签约时间"
            )
        }

        // 2. 检查是否已被玩家签约
        if (isPlayerSignedByPlayer(saveId, prospect.playerId)) {
            return ProspectResult(
                prospectId = prospect.prospectId,
                action = ProspectAction.PATH_INTERRUPTED,
                playerId = prospect.playerId,
                clubId = null,
                butterflyEventTriggered = false,
                message = "已被玩家签约"
            )
        }

        // 3. 查找感兴趣的 AI 俱乐部
        val interestedClubs = findInterestedClubs(prospect, saveId)
        if (interestedClubs.isEmpty()) {
            return ProspectResult(
                prospectId = prospect.prospectId,
                action = ProspectAction.DEFAULT_PATH_EXECUTED,
                playerId = prospect.playerId,
                clubId = null,
                butterflyEventTriggered = false,
                message = "无 AI 俱乐部感兴趣，按默认路径"
            )
        }

        // 4. 按 target_score 排序
        val rankedClubs = interestedClubs.sortedByDescending { it.targetScore }

        // 5. 检查约束：每豪门每窗 ≤2 名 21 岁以下高潜
        val eligibleClub = rankedClubs.find { club ->
            !isClubAtYoungTalentLimit(saveId, club.clubId)
        }

        if (eligibleClub != null) {
            // 6. 执行 AI 签约
            aiTransferEngine.signYoungTalent(eligibleClub.clubId, prospect.playerId, saveId)

            // 7. 更新星状态
            saveDb.prospectStateDao().updateStatus(
                saveId, prospect.prospectId, ProspectStatus.SIGNED_EARLY.name
            )

            return ProspectResult(
                prospectId = prospect.prospectId,
                action = ProspectAction.AI_SIGNED,
                playerId = prospect.playerId,
                clubId = eligibleClub.clubId,
                butterflyEventTriggered = true,  // AI 签约也触发蝴蝶
                message = "${prospect.playerName} 被 ${getClubName(eligibleClub.clubId)} 签约"
            )
        }

        return ProspectResult(
            prospectId = prospect.prospectId,
            action = ProspectAction.DEFAULT_PATH_EXECUTED,
            playerId = prospect.playerId,
            clubId = null,
            butterflyEventTriggered = false,
            message = "AI 俱乐部均达限额，按默认路径"
        )
    }

    private fun isClubAtYoungTalentLimit(saveId: Int, clubId: Int): Boolean {
        val count = saveDb.saveTransferOfferDao()
            .countYoungTalentSignedByClub(saveId, clubId, ageMax = 21, minPa = 80)
        return count >= config.maxYoungTalentPerClubPerWindow  // 默认 2
    }
}
```

## 八、入口服务

```kotlin
// api/HistoricalProspectService.kt
class HistoricalProspectService(
    private val poolManager: ProspectPoolManager,
    private val discoverer: ProspectDiscoverer,
    private val defaultPathExecutor: DefaultPathExecutor,
    private val earlySignInterrupter: EarlySignInterrupter,
    private val aiCompetitorSigner: AiCompetitorSigner
) {

    /**
     * 由 T07 每日推进调用
     */
    suspend fun processDailyCheck(ctx: AdvanceContext): List<ProspectResult> {
        val results = mutableListOf<ProspectResult>()

        // 1. 激活到期新星
        val activated = poolManager.activateProspects(ctx.currentDate, ctx.saveId)

        // 2. 处理默认路径执行
        val activeProspects = poolManager.getActiveProspects(ctx.saveId)
        for (prospect in activeProspects) {
            val result = defaultPathExecutor.executeDefaultPath(
                prospect, ctx.currentDate, ctx.saveId
            )
            if (result.action != ProspectAction.DEFAULT_PATH_EXECUTED ||
                result.message != "未到转会日期") {
                results.add(result)
            }
        }

        // 3. 处理 AI 竞争签约（转会窗期间）
        if (ctx.isTransferWindowOpen) {
            for (prospect in activeProspects) {
                val aiResult = aiCompetitorSigner.processAiSigning(
                    prospect, ctx.currentDate, ctx.saveId
                )
                if (aiResult.action == ProspectAction.AI_SIGNED) {
                    results.add(aiResult)
                }
            }
        }

        return results
    }

    /**
     * 玩家签约球员时调用（检查是否历史新星）
     */
    suspend fun onPlayerSigned(
        playerId: Int,
        signedByClubId: Int,
        currentDate: LocalDate,
        saveId: Int
    ): ProspectResult {
        return earlySignInterrupter.onPlayerEarlySigned(
            playerId, signedByClubId, currentDate, saveId
        )
    }

    /**
     * 球探任务结束时调用
     */
    suspend fun onScoutTaskCompleted(
        saveId: Int,
        scoutId: Int,
        regionCode: String,
        currentDate: LocalDate
    ): List<DiscoveryResult> {
        return discoverer.tryDiscover(saveId, scoutId, regionCode, currentDate)
    }
}
```

## 十一、关键风险与应对

### 1. 新星全部被玩家签走

- **表现**：玩家签下所有历史新星
- **应对**：
  - AI 竞争签约（豪门会抢）
  - 签约难度（高身价+高工资）
  - 球员加盟意愿（可能拒绝玩家）

### 2. 蝴蝶效应级联

- **表现**：一次提前签约触发无数蝴蝶
- **应对**：
  - T20 预算控制（每赛季 ≤20 事件）
  - 单次触发只生成 1 个蝴蝶事件

### 3. 默认路径错误

- **表现**：C 罗没去曼联但系统仍按历史执行
- **应对**：
  - 每次执行前检查球员当前俱乐部
  - 若俱乐部不匹配则视为路径打断
