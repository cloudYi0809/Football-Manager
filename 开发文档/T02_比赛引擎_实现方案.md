## 一、整体架构

### 1. 四层引擎流水线

```
MatchInput（阵容+战术+状态）
        ↓
┌─────────────────────────────────┐
│ Layer 1: RatingLayer  评分层     │  计算双方 attack/defense/control
└─────────────────────────────────┘
        ↓
┌─────────────────────────────────┐
│ Layer 2: XGLayer      xG 生成层  │  强弱映射为预期进球
└─────────────────────────────────┘
        ↓
┌─────────────────────────────────┐
│ Layer 3: PoissonLayer 泊松进球层 │  生成进球数 + 极端比分抑制
└─────────────────────────────────┘
        ↓
┌─────────────────────────────────┐
│ Layer 4: EventLayer   事件生成层 │  18 tick 事件流 + 球星模板
└─────────────────────────────────┘
        ↓
MatchResult（比分+统计+事件+评分）
```

### 1. 算法（严格按 V0.2 04 文档）

```kotlin
// core/RatingLayer.kt
class RatingLayer(private val config: MatchConfig) {

    fun rate(teamSheet: TeamSheet): TeamRating {
        val attack = calculateAttack(teamSheet)
        val defense = calculateDefense(teamSheet)
        val control = calculateControl(teamSheet)
        val morale = calculateMoraleFactor(teamSheet)
        val condition = calculateConditionFactor(teamSheet)
        val tacticalFit = calculateTacticalFit(teamSheet)

        return TeamRating(
            attackScore = attack,
            defenseScore = defense,
            controlScore = control,
            moraleFactor = morale,
            conditionFactor = condition,
            tacticalFit = tacticalFit,
            breakdown = RatingBreakdown(...)
        )
    }

    /**
     * 进攻评分（V0.2 04 §三.2）
     * attack_score = forward_finish * 0.28 + chance_creation * 0.24
     *              + wing_threat * 0.14 + midfield_supply * 0.14
     *              + set_piece_attack * 0.08 + morale * 0.05 + tactical_fit * 0.07
     */
    private fun calculateAttack(sheet: TeamSheet): Double {
        val starters = sheet.starting11
        val forwardFinish = avgByPosition(starters, setOf(Position.ST, Position.CF, Position.LW, Position.RW)) {
            it.attributes.finishing * 0.6 + it.attributes.shotPower * 0.2 + it.attributes.composure * 0.2
        }
        val chanceCreation = avgByPosition(starters, setOf(Position.AM, Position.CM)) {
            it.attributes.vision * 0.5 + it.attributes.passing * 0.3 + it.attributes.technique * 0.2
        }
        val wingThreat = avgByPosition(starters, setOf(Position.LW, Position.RW, Position.LB, Position.RB)) {
            it.attributes.dribbling * 0.5 + it.attributes.pace * 0.3 + it.attributes.technique * 0.2
        }
        val midfieldSupply = avgByPosition(starters, setOf(Position.CM, Position.DM)) {
            it.attributes.passing * 0.6 + it.attributes.vision * 0.4
        }
        val setPieceAttack = avg(starters) {
            it.attributes.technique * 0.5 + it.attributes.vision * 0.5
        }
        val morale = avg(starters) { it.morale.toDouble() } / 100.0
        val tacticalFit = calculateTacticalFit(sheet)

        val weights = config.attackWeights  // 从配置读取
        return forwardFinish * weights.forwardFinish +
               chanceCreation * weights.chanceCreation +
               wingThreat * weights.wingThreat +
               midfieldSupply * weights.midfieldSupply +
               setPieceAttack * weights.setPieceAttack +
               morale * weights.morale +
               tacticalFit * weights.tacticalFit
    }

    /**
     * 防守评分（V0.2 04 §三.3）
     */
    private fun calculateDefense(sheet: TeamSheet): Double {
        val starters = sheet.starting11
        val centerBack = avgByPosition(starters, setOf(Position.CB)) {
            it.attributes.marking * 0.4 + it.attributes.tackling * 0.3 + it.attributes.interceptions * 0.3
        }
        val goalkeeper = avgByPosition(starters, setOf(Position.GK)) {
            it.attributes.gkDiving * 0.3 + it.attributes.gkPositioning * 0.3 +
            it.attributes.gkReflexes * 0.2 + it.attributes.gkHandling * 0.2
        }
        val defensiveMid = avgByPosition(starters, setOf(Position.DM)) {
            it.attributes.interceptions * 0.5 + it.attributes.tackling * 0.3 + it.attributes.standingTackle * 0.2
        }
        val fullback = avgByPosition(starters, setOf(Position.LB, Position.RB)) {
            it.attributes.tackling * 0.4 + it.attributes.marking * 0.3 + it.attributes.pace * 0.3
        }
        val defensiveShape = calculateDefensiveShape(sheet)
        val setPieceDefense = avg(starters) {
            it.attributes.marking * 0.6 + it.attributes.heading * 0.4
        }
        val tacticalFit = calculateTacticalFit(sheet)
        val condition = avg(starters) { it.condition.toDouble() } / 100.0

        val w = config.defenseWeights
        return centerBack * w.centerBack +
               goalkeeper * w.goalkeeper +
               defensiveMid * w.defensiveMid +
               fullback * w.fullback +
               defensiveShape * w.defensiveShape +
               setPieceDefense * w.setPieceDefense +
               tacticalFit * w.tacticalFit +
               condition * w.condition
    }

    /**
     * 中场控制评分（V0.2 04 §三.4）
     */
    private fun calculateControl(sheet: TeamSheet): Double {
        val starters = sheet.starting11
        val passing = avg(starters) { it.attributes.passing.toDouble() }
        val technique = avg(starters) { it.attributes.technique.toDouble() }
        val vision = avg(starters) { it.attributes.vision.toDouble() }
        val workRate = avg(starters) { it.attributes.workRate.toDouble() }
        val pressing = avg(starters) { it.attributes.pressing.toDouble() }
        val teamwork = avg(starters) { it.attributes.teamwork.toDouble() }
        val tacticalFit = calculateTacticalFit(sheet)

        val w = config.controlWeights
        return passing * w.passing +
               technique * w.technique +
               vision * w.vision +
               workRate * w.workRate +
               pressing * w.pressing +
               teamwork * w.teamwork +
               tacticalFit * w.tacticalFit
    }

    // 位置权重表（V0.2 04 §三.1）
    private fun getPositionWeight(pos: Position): PositionWeights = when (pos) {
        Position.ST, Position.CF -> PositionWeights(attack = 0.70, midfield = 0.10, defense = 0.05)
        Position.LW, Position.RW -> PositionWeights(0.55, 0.20, 0.10)
        Position.AM              -> PositionWeights(0.45, 0.35, 0.05)
        Position.CM              -> PositionWeights(0.25, 0.45, 0.20)
        Position.DM              -> PositionWeights(0.10, 0.35, 0.40)
        Position.LB, Position.RB -> PositionWeights(0.20, 0.20, 0.40)
        Position.CB              -> PositionWeights(0.05, 0.10, 0.70)
        Position.GK              -> PositionWeights(0.0, 0.0, 0.85)
    }

    private fun calculateTacticalFit(sheet: TeamSheet): Double {
        // 阵型与战术匹配度：0.90 - 1.10
        // 例如 433 + 高位压迫 = 1.08，442 + 防守反击 = 1.05
        return config.formationTacticFitMatrix
            .getOrDefault("${sheet.formation}_${sheet.tactic.style}", 1.0)
    }

    // 工具方法
    private fun avgByPosition(players: List<PlayerState>, positions: Set<Position>, selector: (PlayerState) -> Double): Double {
        val filtered = players.filter { it.position in positions }
        if (filtered.isEmpty()) return 50.0  // 默认值
        return filtered.sumOf(selector) / filtered.size
    }

    private fun avg(players: List<PlayerState>, selector: (PlayerState) -> Double): Double =
        players.sumOf(selector) / players.size
}
```

### 1. 算法（V0.2 04 §五）

```kotlin
// core/XGLayer.kt
class XGLayer(private val config: MatchConfig) {

    fun generate(homeRating: TeamRating, awayRating: TeamRating, ctx: CompetitionContext): XGResult {
        // 1. 基础 xG（V0.2 04 §五.1）
        val homeBase = config.leagueAvgHomeXg    // 1.45
        val awayBase = config.leagueAvgAwayXg    // 1.15

        // 2. 强弱修正（V0.2 04 §五.2）
        val homeAttackDefRatio = homeRating.attackScore / awayRating.defenseScore
        val awayAttackDefRatio = awayRating.attackScore / homeRating.defenseScore
        val homeControlRatio = homeRating.controlScore / awayRating.controlScore
        val awayControlRatio = awayRating.controlScore / homeRating.controlScore

        val homeTacticalMod = calculateTacticalModifier(homeRating, awayRating)
        val awayTacticalMod = calculateTacticalModifier(awayRating, homeRating)

        val homeMoraleMod = homeRating.moraleFactor
        val awayMoraleMod = awayRating.moraleFactor
        val homeCondMod = homeRating.conditionFactor
        val awayCondMod = awayRating.conditionFactor

        val homeImportanceMod = calculateImportanceModifier(ctx, isHome = true)
        val awayImportanceMod = calculateImportanceModifier(ctx, isHome = false)

        // 3. xG 计算（V0.2 04 §五.2 公式）
        val homeXg = homeBase *
            clamp(homeAttackDefRatio, 0.55, 1.75) *
            clamp(0.85 + (homeControlRatio - 1.0) * 0.25, 0.80, 1.20) *
            homeTacticalMod *
            homeMoraleMod *
            homeCondMod *
            homeImportanceMod

        val awayXg = awayBase *
            clamp(awayAttackDefRatio, 0.55, 1.75) *
            clamp(0.85 + (awayControlRatio - 1.0) * 0.25, 0.80, 1.20) *
            awayTacticalMod *
            awayMoraleMod *
            awayCondMod *
            awayImportanceMod

        // 4. 上下限（V0.2 04 §五.3）
        val isExtreme = isExtremeMatch(homeRating, awayRating)
        val maxXg = if (isExtreme) config.maxExtremeXg else config.maxRegularXg  // 7.0 vs 4.5

        val finalHomeXg = homeXg.coerceIn(config.minXg, maxXg)  // min=0.15
        val finalAwayXg = awayXg.coerceIn(config.minXg, maxXg)

        return XGResult(
            homeXg = finalHomeXg,
            awayXg = finalAwayXg,
            homeModifiers = XgModifiers(homeAttackDefRatio, homeControlRatio, homeTacticalMod, homeMoraleMod, homeCondMod, homeImportanceMod),
            awayModifiers = XgModifiers(awayAttackDefRatio, awayControlRatio, awayTacticalMod, awayMoraleMod, awayCondMod, awayImportanceMod),
            isExtremeMatch = isExtreme
        )
    }

    /**
     * 极端强弱差检测
     * V0.2 04 §五.3："极端强弱差可突破，但需触发 rare_extreme_match 标记"
     */
    private fun isExtremeMatch(home: TeamRating, away: TeamRating): Boolean {
        val ratingDiff = abs(home.attackScore + home.defenseScore - away.attackScore - away.defenseScore)
        return ratingDiff > config.extremeMatchThreshold  // 默认 30
    }

    private fun calculateTacticalModifier(my: TeamRating, opp: TeamRating): Double {
        // 战术克制综合修正
        return 1.0  // 由 CounterMatrix 注入
    }

    private fun calculateImportanceModifier(ctx: CompetitionContext, isHome: Boolean): Double {
        // 比赛重要性修正：杯赛决赛、德比、保级战
        return when (ctx.importance) {
            Importance.FINAL -> 1.05
            Importance.DERBY -> 1.03
            Importance.RELEGATION_BATTLE -> 1.04
            Importance.NORMAL -> 1.0
        }
    }

    private fun clamp(v: Double, min: Double, max: Double): Double = v.coerceIn(min, max)
}
```

### 1. 算法（V0.2 04 §六）

```kotlin
// core/PoissonLayer.kt
class PoissonLayer(private val config: MatchConfig) {

    private val random = SplittableRandom()  // 可注入种子

    fun generateGoals(homeXg: Double, awayXg: Double, isExtreme: Boolean): Pair<Int, Int> {
        // 1. 泊松生成（V0.2 04 §六）
        var homeGoals = poissonSample(homeXg)
        var awayGoals = poissonSample(awayXg)

        // 2. 极端比分抑制（V0.2 04 §六）
        if (!isExtreme) {
            homeGoals = applyExtremeDampening(homeGoals)
            awayGoals = applyExtremeDampening(awayGoals)
        }

        return homeGoals to awayGoals
    }

    /**
     * 泊松采样：Knuth 算法
     */
    private fun poissonSample(lambda: Double): Int {
        val L = exp(-lambda)
        var k = 0
        var p = 1.0
        do {
            k++
            p *= random.nextDouble()
        } while (p > L)
        return k - 1
    }

    /**
     * 极端比分抑制（V0.2 04 §六）
     * - 单队 5 球以上概率乘 0.65
     * - 单队 6 球以上概率乘 0.40
     * - 单队 7 球以上概率乘 0.20
     */
    private fun applyExtremeDampening(goals: Int): Int {
        if (goals < 5) return goals
        val dampenProb = when (goals) {
            5 -> 0.65
            6 -> 0.40
            else -> 0.20
        }
        // 概率性抑制：以 (1 - dampenProb) 概率保留，否则减 1
        return if (random.nextDouble() < dampenProb) goals - 1 else goals
    }
}
```

```kotlin
// core/EventLayer.kt
class EventLayer(
    private val config: MatchConfig,
    private val starTemplateRegistry: StarTemplateRegistry
) {

    /**
     * 生成 18 个常规 tick + 1-2 个补时 tick
     * V0.2 04 §七："每 5 分钟一个 tick，共 18 个常规 tick + 补时 tick"
     */
    fun generateEvents(
        input: MatchInput,
        homeRating: TeamRating,
        awayRating: TeamRating,
        xg: XGResult,
        homeGoals: Int,
        awayGoals: Int
    ): List<MatchEvent> {
        val events = mutableListOf<MatchEvent>()
        val totalTicks = 18 + random補時(1, 2)  // 18-20 个 tick
        val pendingHomeGoals = homeGoals
        val pendingAwayGoals = awayGoals

        // 预分配进球时刻
        val homeGoalMinutes = distributeGoalMinutes(pendingHomeGoals, isHome = true)
        val awayGoalMinutes = distributeGoalMinutes(pendingAwayGoals, isHome = false)

        for (tick in 0 until totalTicks) {
            val minute = tick * 5 + random.nextInt(5)
            val isStoppageTime = tick >= 18

            // 控球方判定（基于 control_score + 当前比分 + 战术倾向）
            val possessionSide = determinePossession(homeRating, awayRating, homeGoals, awayGoals)

            // 事件类型抽样
            val eventType = sampleEventType(possessionSide, homeRating, awayRating, input)

            // 选择参与球员
            val actingPlayer = selectPlayerForEvent(eventType, possessionSide, input)

            // 处理事件结果
            val event = resolveEvent(eventType, minute, possessionSide, actingPlayer, input)
            events.add(event)

            // 进球时刻触发
            if (minute in homeGoalMinutes) events.add(createGoalEvent(minute, TeamSide.HOME, input))
            if (minute in awayGoalMinutes) events.add(createGoalEvent(minute, TeamSide.AWAY, input))

            // 红黄牌/伤病检查
            events.addAll(checkCardsAndInjuries(minute, input, homeRating, awayRating))
        }

        return events.sortedBy { it.minute }
    }

    /**
     * 事件类型基础概率（V0.2 04 §七 事件概率基础表）
     */
    private fun sampleEventType(
        possessionSide: TeamSide,
        homeRating: TeamRating,
        awayRating: TeamRating,
        input: MatchInput
    ): EventType {
        val baseProbs = config.eventBaseProbabilities  // 10 类事件的基础概率
        val tacticalMods = calculateTacticalEventModifiers(input, possessionSide)
        val scoreMods = calculateScoreEventModifiers(input)  // 落后方更激进

        val adjustedProbs = baseProbs.mapValues { (type, prob) ->
            prob * (tacticalMods[type] ?: 1.0) * (scoreMods[type] ?: 1.0)
        }

        return weightedSample(adjustedProbs)
    }

    /**
     * 球星模板影响事件权重
     * V0.2 04 §九：球星模板不直接加进球，只改事件权重
     */
    private fun applyStarTemplateModifiers(
        player: PlayerState,
        eventProbs: MutableMap<EventType, Double>
    ) {
        val template = player.starTemplate ?: return
        val starMod = starTemplateRegistry.get(template)

        // 大罗型：shot_conversion +8%, high_quality_chance_creation_self +6%
        // 卡卡型：central_progression +8%, through_ball_event +7%
        // ... 见 V0.2 04 §九
        starMod.modifiers.forEach { (eventType, mod) ->
            eventProbs[eventType] = (eventProbs[eventType] ?: 0.0) * (1.0 + mod)
        }
    }
}
```

```kotlin
// core/CalibrationLayer.kt
class CalibrationLayer {

    /**
     * 赛后球员评分（V0.2 04 §十一）
     * match_rating = 6.5 + goals * pos_weight + assists * 0.6 + key_passes * 0.08
     *              + tackles_success * 0.05 + interceptions * 0.05 + saves * 0.10
     *              - errors_leading_to_goal * 0.8 - red_card * 1.5 - own_goal * 1.0
     *              + team_result_bonus
     * 限制：4.0 - 10.0
     */
    fun calculatePlayerRatings(
        events: List<MatchEvent>,
        homeScore: Int,
        awayScore: Int,
        startingPlayers: List<PlayerState>
    ): Map<String, Double> {
        val ratings = mutableMapOf<String, Double>()

        for (player in startingPlayers) {
            val playerEvents = events.filter { it.playerId == player.id }
            val goals = playerEvents.count { it.type == EventType.GOAL }
            val assists = playerEvents.count { it.secondaryPlayerId == player.id && it.type == EventType.GOAL }
            val keyPasses = playerEvents.count { it.type == EventType.DANGEROUS_ATTACK }
            val tackles = playerEvents.count { it.type == EventType.FOUL }.let { 0 } // 占位
            val interceptions = playerEvents.count { it.type == EventType.CLEARANCE }
            val saves = if (player.position == Position.GK) playerEvents.count { it.type == EventType.SAVE } else 0
            val errors = 0  // 简化
            val redCard = playerEvents.count { it.type == EventType.RED_CARD }
            val ownGoal = 0

            val posWeight = when (player.position) {
                Position.ST, Position.CF, Position.LW, Position.RW -> 1.0
                Position.AM, Position.CM -> 0.7
                Position.DM -> 0.4
                Position.LB, Position.RB, Position.CB -> 0.2
                Position.GK -> 0.0
            }

            val teamResultBonus = calculateTeamBonus(player, homeScore, awayScore)

            var rating = 6.5 +
                goals * posWeight +
                assists * 0.6 +
                keyPasses * 0.08 +
                tackles * 0.05 +
                interceptions * 0.05 +
                saves * 0.10 -
                errors * 0.8 -
                redCard * 1.5 -
                ownGoal * 1.0 +
                teamResultBonus

            rating = rating.coerceIn(4.0, 10.0)
            ratings[player.id] = rating
        }

        return ratings
    }
}
```

```kotlin
// api/MatchSimulator.kt
class MatchSimulator(
    private val ratingLayer: RatingLayer,
    private val xgLayer: XGLayer,
    private val poissonLayer: PoissonLayer,
    private val eventLayer: EventLayer,
    private val calibrationLayer: CalibrationLayer
) {

    fun simulate(input: MatchInput): MatchResult {
        // 1. 评分层
        val homeRating = ratingLayer.rate(input.homeTeam)
        val awayRating = ratingLayer.rate(input.awayTeam)

        // 2. xG 层
        val xg = xgLayer.generate(homeRating, awayRating, input.competition)

        // 3. 泊松进球层
        val (homeGoals, awayGoals) = poissonLayer.generateGoals(xg.homeXg, xg.awayXg, xg.isExtremeMatch)

        // 4. 事件层
        val events = eventLayer.generateEvents(input, homeRating, awayRating, xg, homeGoals, awayGoals)

        // 5. 校准层：评分 + 统计
        val playerRatings = calibrationLayer.calculatePlayerRatings(
            events, homeGoals, awayGoals, input.homeTeam.starting11 + input.awayTeam.starting11
        )
        val homeStats = calculateStats(events, TeamSide.HOME)
        val awayStats = calculateStats(events, TeamSide.AWAY)
        val injuries = extractInjuries(events)
        val cards = extractCards(events)

        return MatchResult(
            matchId = input.matchId,
            homeScore = homeGoals,
            awayScore = awayGoals,
            homeXg = xg.homeXg,
            awayXg = xg.awayXg,
            events = events,
            playerRatings = playerRatings,
            homeStats = homeStats,
            awayStats = awayStats,
            injuries = injuries,
            cards = cards
        )
    }
}
```

## 十二、关键风险与应对

### 1. 评分公式偏差

- **表现**：强弱队评分差 < 15，导致 xG 失真
- **应对**：W4 D5 单测验证评分差，不达标先调位置权重或属性聚合方式
- **降级**：若 V0.2 公式偏差大，可临时调整 attack_weights 配置（不改架构）

### 2. 泊松采样极端值

- **表现**：5+ 球场次过多或过少
- **应对**：先调 extreme_score_dampening 的 multiplier（0.65/0.40/0.20）
- **红线**：不改泊松算法本身，只调抑制系数

### 3. 事件层与进球不匹配

- **表现**：泊松生成 3 球但事件层只产生 1 个进球事件
- **应对**：进球时刻预分配机制（见 EventLayer），事件层只负责"叙事"不负责"决定比分"
- **原则**：比分以泊松层为准，事件层补全叙事

### 4. 球星模板过强

- **表现**：球星包揽进球，违反 V0.2 §九"球星模板不直接加进球"
- **应对**：严格只改事件权重，不改进球转化率（shot_conversion_bonus 仅微调）
- **测试**：跑 1000 场，球星进球占比应 < 30%
