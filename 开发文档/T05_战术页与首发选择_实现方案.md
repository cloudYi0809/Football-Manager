// validation/Starting11Validator.kt
class Starting11Validator {

    /**
     * 校验首发 11 人
     */
    fun validate(starting11: List<PlayerSlot>): ValidationResult {
        val errors = mutableListOf<String>()

        // 1. 必须 11 人
        if (starting11.size != 11) {
            errors.add("首发必须 11 人，当前 ${starting11.size}")
        }

        // 2. 必须有 1 名 GK
        val gkCount = starting11.count { it.position == Position.GK }
        if (gkCount != 1) {
            errors.add("首发必须有且仅有 1 名门将，当前 $gkCount")
        }

        // 3. 球员不重复
        val playerIds = starting11.mapNotNull { it.playerId }
        if (playerIds.size != playerIds.toSet().size) {
            errors.add("首发存在重复球员")
        }

        // 4. 无伤病
        // 5. 无停赛
        // 这两项需读取球员状态，由调用方传入

        return ValidationResult(errors.isEmpty(), errors)
    }
}

data class ValidationResult(val isValid: Boolean, val errors: List<String>)

// validation/PositionFitChecker.kt
class PositionFitChecker {

    /**
     * 计算球员在特定位置的适配度（0-1）
     * 基于球员主位置/副位置
     */
    fun calculateFit(player: SavePlayerStateWithAttributes, targetPosition: Position): Double {
        // 主位置完全匹配
        if (player.primaryPosition == targetPosition) return 1.0

        // 副位置匹配
        val secondaryPositions = parseSecondaryPositions(player.secondaryPositions)
        if (targetPosition in secondaryPositions) return 0.85

        // 同类位置（如 CB→DM，CM→AM）
        val positionGroup = getPositionGroup(targetPosition)
        val playerGroup = getPositionGroup(player.primaryPosition)
        if (positionGroup == playerGroup) return 0.70

        // 完全不匹配
        return 0.40
    }

    private fun getPositionGroup(pos: Position): PositionGroup = when (pos) {
        Position.GK -> PositionGroup.GK
        Position.CB, Position.LB, Position.RB -> PositionGroup.DEFENSE
        Position.DM, Position.CM, Position.AM -> PositionGroup.MIDFIELD
        Position.LW, Position.RW, Position.ST, Position.CF -> PositionGroup.ATTACK
    }

    private fun parseSecondaryPositions(s: String?): Set<Position> {
        if (s.isNullOrBlank()) return emptySet()
        return s.split(",").mapNotNull { name ->
            runCatching { Position.valueOf(name.trim()) }.getOrNull()
        }.toSet()
    }
}

enum class PositionGroup { GK, DEFENSE, MIDFIELD, ATTACK }

// validation/TacticalProficiencyCalculator.kt
class TacticalProficiencyCalculator(private val config: TacticConfig) {

    /**
     * 战术熟练度（0-100）
     * V0.1 03 §3："改变战术参数实时更新战术熟练度提示"
     */
    fun calculate(
        style: TacticalStyle,
        parameters: TacticalParameters,
        starting11: List<PlayerSlot>
    ): TacticalProficiency {
        // 1. 阵型适配
        val formationFit = calculateFormationFit(style, starting11)

        // 2. 球员属性匹配
        val playerAttributeFit = calculatePlayerAttributeFit(style, starting11)

        // 3. 参数合理性
        val parameterReasonableness = calculateParameterReasonableness(style, parameters)

        val overall = (formationFit * 0.35 + playerAttributeFit * 0.45 + parameterReasonableness * 0.20) * 100

        val risk = calculateRisk(parameters, starting11)

        return TacticalProficiency(
            overall = overall.toInt(),
            formationFit = formationFit,
            playerAttributeFit = playerAttributeFit,
            parameterReasonableness = parameterReasonableness,
            riskLevel = risk
        )
    }

    /**
     * 体能风险提示
     * V0.1 03 §3："战术过于激进时提示体能风险"
     */
    private fun calculateRisk(
        parameters: TacticalParameters,
        starting11: List<PlayerSlot>
    ): RiskLevel {
        val aggressionScore = parameters.tempo + parameters.pressingIntensity + parameters.defensiveLine
        return when {
            aggressionScore >= 25 -> RiskLevel.HIGH
            aggressionScore >= 18 -> RiskLevel.MEDIUM
            else -> RiskLevel.LOW
        }
    }
}

data class TacticalProficiency(
    val overall: Int,
    val formationFit: Double,
    val playerAttributeFit: Double,
    val parameterReasonableness: Double,
    val riskLevel: RiskLevel
)

enum class RiskLevel { LOW, MEDIUM, HIGH }

                    onDragStart = { offset ->
                        // 识别拖拽起点对应的 slotId
                    },
                    onDragEnd = { /* 完成 */ },
                    onDrag = { change, dragAmount ->
                        // 更新拖拽位置
                    }
                )
            }
    ) {
        // 1. 绘制球场背景（绿色 + 草地纹理）
        drawRect(color = Color(0xFF4CAF50))

        // 2. 绘制球场线条（中圈、禁区、球门）
        drawPitchLines()

        // 3. 绘制 11 个位置点
        formation.positions.forEach { pos ->
            val pixelX = size.width * pos.x
            val pixelY = size.height * pos.y
            val slot = starting11.find { it.slotId == pos.slotId }

            drawCircle(
                color = if (slot?.playerId != null) Color.Yellow else Color.White,
                radius = 24f,
                center = Offset(pixelX, pixelY)
            )

            // 绘制球员号码/位置缩写
            slot?.let {
                drawText(
                    text = getPlayerDisplayText(it),
                    x = pixelX,
                    y = pixelY
                )
            }
        }
    }
}
