### 2. 活跃范围机制（V0.2 性能关键）

```
全量模拟（活跃范围）              简化模拟（非活跃范围）
─────────────────────              ─────────────────────
玩家所在联赛的所有俱乐部            其他联赛俱乐部
玩家俱乐部 + 同联赛 19 队           只更新积分/排名/关键转会
深度模拟：训练/伤病/士气/转会       浅度模拟：月度汇总
每日执行                            每月或转会窗执行
```

**性能预算**（P95 ≤3 秒）：
- 玩家俱乐部深度模拟：200ms
- 同联赛 19 队活跃模拟：1500ms
- 非活跃联赛简化模拟：300ms
- 比赛+事件+新闻：500ms
- 数据库事务+存档：500ms
- **总计：约 3000ms**

```kotlin
// api/DailyAdvanceScheduler.kt
class DailyAdvanceScheduler(
    private val saveDb: SaveDatabase,
    private val historyDb: HistoryDatabase,
    private val dailyExecutor: DailyTaskExecutor,
    private val weeklyExecutor: WeeklyTaskExecutor,
    private val monthlyExecutor: MonthlyTaskExecutor,
    private val seasonEndExecutor: SeasonEndExecutor,
    private val activeScopeManager: ActiveScopeManager,
    private val rollbackService: AdvanceRollback,
    private val config: ProgressionConfig,
    private val perfLogger: PerfLogger
) {

    /**
     * 推进一天
     * V0.1 11 §十："每日推进前可自动保存。如果推进失败，回滚到上一日。"
     */
    suspend fun advanceOneDay(): AdvanceResult {
        val startTime = System.currentTimeMillis()

        // 1. 创建推进前快照（用于失败回滚）
        val snapshot = rollbackService.createPreAdvanceSnapshot()

        // 2. 构建推进上下文
        val context = buildAdvanceContext()

        return try {
            // 3. 执行每日任务（13 项）
            val dailyEvents = dailyExecutor.execute(context)

            // 4. 每周任务（若周一）
            val weeklyEvents = if (context.isWeekStart) {
                weeklyExecutor.execute(context)
            } else emptyList()

            // 5. 每月任务（若月初）
            val monthlyEvents = if (context.isMonthStart) {
                monthlyExecutor.execute(context)
            } else emptyList()

            // 6. 赛季结束任务（若赛季最后一日）
            val seasonEvents = if (context.isSeasonEnd) {
                seasonEndExecutor.execute(context)
            } else emptyList()

            // 7. 聚合结果
            val allEvents = dailyEvents + weeklyEvents + monthlyEvents + seasonEvents
            val matches = extractMatches(allEvents)
            val news = extractNews(allEvents)
            val todos = refreshTodos(context)

            // 8. 更新存档当前日期
            saveDb.saveWorldStateDao().updateCurrentDate(context.saveId, context.nextDate.toString())
            saveDb.saveManifestDao().updateCurrentDate(getSaveUuid(), context.nextDate.toString())

            // 9. 自动保存（V0.1 11 §十）
            if (config.autoSaveEnabled) {
                saveCheckpointIfNeeded(context)
            }

            // 10. 性能记录
            val duration = System.currentTimeMillis() - startTime
            perfLogger.log("daily_advance", duration, context.saveId)

            // 11. 性能告警
            if (duration > config.perfWarningMs) {
                perfLogger.warn("daily_advance P95 超时：${duration}ms")
            }

            AdvanceResult(
                newDate = context.nextDate,
                events = allEvents,
                matches = matches,
                news = news,
                todos = todos,
                durationMs = duration
            )

        } catch (e: Exception) {
            // 回滚到上一日
            rollbackService.rollback(snapshot)
            AdvanceResult(
                newDate = context.currentDate,
                events = emptyList(),
                matches = emptyList(),
                news = emptyList(),
                todos = emptyList(),
                durationMs = System.currentTimeMillis() - startTime,
                rollbackReason = "推进失败：${e.message}"
            )
        }
    }

    /**
     * 推进多天（玩家可一次推进到下一比赛日/月底/转会窗结束）
     */
    suspend fun advanceUntil(targetDate: LocalDate): List<AdvanceResult> {
        val results = mutableListOf<AdvanceResult>()
        var currentDate = getCurrentDate()

        while (currentDate.isBefore(targetDate)) {
            val result = advanceOneDay()
            results.add(result)

            // 若中途回滚，停止推进
            if (result.rollbackReason != null) break

            currentDate = result.newDate
        }

        return results
    }

    /**
     * 推进到下一比赛日（最常用，玩家点"下一场"）
     */
    suspend fun advanceToNextMatch(): AdvanceResult {
        val nextMatchDate = historyDb.matchDao()
            .getNextMatchDateForClub(getManagerClubId(), getCurrentSeasonId(), getCurrentDate())

        return advanceUntil(nextMatchDate).last()
    }

    private fun buildAdvanceContext(): AdvanceContext {
        val currentDate = getCurrentDate()
        val nextDate = currentDate.plusDays(1)

        return AdvanceContext(
            saveId = getCurrentSaveId(),
            currentDate = currentDate,
            nextDate = nextDate,
            currentSeasonId = getCurrentSeasonId(),
            managerClubId = getManagerClubId(),
            activeLeagueIds = activeScopeManager.getActiveLeagueIds(),
            isTransferWindowOpen = isTransferWindow(currentDate),
            isMatchDay = isMatchDayForManager(currentDate),
            isWeekStart = nextDate.dayOfWeek == DayOfWeek.MONDAY,
            isMonthStart = nextDate.dayOfMonth == 1,
            isSeasonEnd = isSeasonEndDate(nextDate),
            randomSeed = currentDate.toEpochDay()
        )
    }
}
```

```kotlin
class DailyTaskExecutor(
    private val trainingTask: TrainingTask,
    private val conditionTask: ConditionTask,
    private val injuryRecoveryTask: InjuryRecoveryTask,
    private val moraleTask: MoraleTask,
    private val scoutTask: ScoutTaskProgress,
    private val youthGrowthTask: YouthGrowthTask,
    private val transferOfferTask: TransferOfferTask,
    private val aiClubTask: AiClubTask,
    private val newsTask: NewsTask,
    private val matchCheckTask: MatchCheckTask,
    private val historyEventTask: HistoryEventTask,
    private val todoRefreshTask: TodoRefreshTask,
    private val activeScopeManager: ActiveScopeManager
) {

    /**
     * 执行 13 项每日任务（V0.1 11 §二.1）
     * 严格按顺序，确保依赖关系正确
     */
    suspend fun execute(ctx: AdvanceContext): List<AdvanceEvent> {
        val events = mutableListOf<AdvanceEvent>()

        // 1. 日期 +1（在 scheduler 已完成）

        // 2. 训练结算（仅活跃范围俱乐部）
        events += trainingTask.execute(ctx)

        // 3. 体能恢复或下降（活跃范围）
        events += conditionTask.execute(ctx)

        // 4. 伤病恢复（活跃范围）
        events += injuryRecoveryTask.execute(ctx)

        // 5. 球员士气变化（活跃范围）
        events += moraleTask.execute(ctx)

        // 6. 球探任务推进（仅玩家俱乐部）
        events += scoutTaskProgress.execute(ctx)

        // 7. 青训成长（活跃范围，每日轻量，月结重）
        events += youthGrowthTask.execute(ctx)

        // 8. 转会报价处理（仅转会窗开启时）
        if (ctx.isTransferWindowOpen) {
            events += transferOfferTask.execute(ctx)
        }

        // 9. AI 俱乐部行动（活跃范围深度，其他范围浅）
        events += aiClubTask.execute(ctx)

        // 10. 新闻生成
        events += newsTask.execute(ctx)

        // 11. 比赛检查（关键，若今日有比赛则调用引擎）
        events += matchCheckTask.execute(ctx)

        // 12. 历史事件检查（V0.2 蝴蝶事件触发）
        events += historyEventTask.execute(ctx)

        // 13. 待办事项刷新
        todoRefreshTask.execute(ctx)

        return events
    }
}
```

### 1. 比赛检查任务（关键任务）

```kotlin
// daily/MatchCheckTask.kt
class MatchCheckTask(
    private val historyDb: HistoryDatabase,
    private val saveDb: SaveDatabase,
    private val matchSimulator: MatchSimulator,       // T02
    private val aiSimulator: AiSimulator,             // T18
    private val activeScopeManager: ActiveScopeManager
) {

    /**
     * V0.1 11 §二.1.11 比赛检查
     * - 玩家俱乐部有比赛 → 调用 T02 引擎模拟
     * - 活跃范围其他俱乐部有比赛 → 调用引擎（结果简化）
     * - 非活跃范围 → 只生成比分，不调用完整引擎
     */
    suspend fun execute(ctx: AdvanceContext): List<AdvanceEvent> {
        val events = mutableListOf<AdvanceEvent>()
        val todayMatches = historyDb.matchDao().getMatchesByDate(ctx.currentDate, ctx.currentSeasonId)

        for (match in todayMatches) {
            val homeDepth = activeScopeManager.getClubDepth(match.homeClubId, ctx)
            val awayDepth = activeScopeManager.getClubDepth(match.awayClubId, ctx)

            val result = when {
                // 玩家比赛：完整模拟 + 完整事件
                match.homeClubId == ctx.managerClubId || match.awayClubId == ctx.managerClubId -> {
                    simulateFullMatch(match, ctx)
                }
                // 活跃范围比赛：完整模拟 + 简化事件
                homeDepth == ClubSimulationDepth.ACTIVE || awayDepth == ClubSimulationDepth.ACTIVE -> {
                    simulateActiveMatch(match, ctx)
                }
                // 非活跃范围：快速比分生成（不调用完整引擎）
                else -> {
                    simulateQuickScore(match, ctx)
                }
            }

            // 更新积分榜
            updateStandings(match, result, ctx)

            // 更新球员统计（活跃范围）
            if (homeDepth != ClubSimulationDepth.MINIMAL || awayDepth != ClubSimulationDepth.MINIMAL) {
                updatePlayerStats(match, result, ctx)
            }

            // 伤病/红黄牌处理（活跃范围）
            if (homeDepth != ClubSimulationDepth.LIGHT && awayDepth != ClubSimulationDepth.LIGHT) {
                processMatchConsequences(match, result, ctx)
            }

            events.add(AdvanceEvent(
                type = AdvanceEventType.MATCH_PLAYED,
                description = "${match.homeClubName} ${result.homeScore}-${result.awayScore} ${match.awayClubName}",
                clubId = null,
                playerId = null,
                priority = if (match.homeClubId == ctx.managerClubId || match.awayClubId == ctx.managerClubId)
                    EventPriority.URGENT else EventPriority.LOW
            ))
        }

        return events
    }

    /**
     * 玩家比赛：完整模拟
     */
    private suspend fun simulateFullMatch(match: MatchEntity, ctx: AdvanceContext): MatchResult {
        val homeSheet = buildTeamSheet(match.homeClubId, ctx)
        val awaySheet = buildTeamSheet(match.awayClubId, ctx)
        val input = MatchInput(
            matchId = match.matchId.toString(),
            homeTeam = homeSheet,
            awayTeam = awaySheet,
            competition = CompetitionContext(match.competitionId),
            randomSeed = ctx.randomSeed + match.matchId
        )
        return matchSimulator.simulate(input)
    }

    /**
     * 活跃范围比赛：完整模拟但事件简化
     */
    private suspend fun simulateActiveMatch(match: MatchEntity, ctx: AdvanceContext): MatchResult {
        // 同样调用引擎，但可省略详细事件生成（V1 简化）
        return simulateFullMatch(match, ctx)
    }

    /**
     * 非活跃范围：快速比分生成
     * 不调用完整引擎，基于评分直接生成比分
     */
    private fun simulateQuickScore(match: MatchEntity, ctx: AdvanceContext): MatchResult {
        val homeRating = getClubOverallRating(match.homeClubId, ctx)
        val awayRating = getClubOverallRating(match.awayClubId, ctx)

        // 简化泊松：基于评分差直接采样
        val homeXg = 1.45 * (homeRating / awayRating).coerceIn(0.55, 1.75)
        val awayXg = 1.15 * (awayRating / homeRating).coerceIn(0.55, 1.75)

        val homeGoals = poissonSample(homeXg, ctx.randomSeed + match.matchId)
        val awayGoals = poissonSample(awayXg, ctx.randomSeed + match.matchId * 2)

        return MatchResult(
            matchId = match.matchId.toString(),
            homeScore = homeGoals,
            awayScore = awayGoals,
            homeXg = homeXg,
            awayXg = awayXg,
            events = emptyList(),         // 不生成事件
            playerRatings = emptyMap(),   // 不计算球员评分
            homeStats = TeamMatchStats(),
            awayStats = TeamMatchStats(),
            injuries = emptyList(),
            cards = emptyList()
        )
    }
}
```

### 2. 训练结算任务

```kotlin
// daily/TrainingTask.kt
class TrainingTask(
    private val saveDb: SaveDatabase,
    private val config: ProgressionConfig
) {

    /**
     * V0.1 11 §二.1.2 训练结算
     * 活跃范围俱乐部每日训练，影响：
     * - 球员训练进度（积累到月结成长）
     * - 体能消耗
     * - 训练质量评估
     */
    suspend fun execute(ctx: AdvanceContext): List<AdvanceEvent> {
        val events = mutableListOf<AdvanceEvent>()

        // 仅活跃范围俱乐部训练
        val activeClubs = saveDb.saveClubStateDao().getByLeagues(ctx.saveId, ctx.activeLeagueIds)
        for (club in activeClubs) {
            if (ctx.isMatchDay && club.clubId == ctx.managerClubId) continue  // 比赛日不训练

            val trainingQuality = calculateTrainingQuality(club, ctx)
            val players = saveDb.savePlayerStateDao().getByClub(ctx.saveId, club.clubId)

            for (player in players) {
                if (player.injuryStatus != "healthy") continue

                // 积累训练进度（月结时兑现）
                val progress = calculateDailyProgress(player, trainingQuality)
                saveDb.savePlayerStateDao().addTrainingProgress(player.id, progress)

                // 体能消耗
                val conditionLoss = calculateConditionLoss(player, trainingQuality, club)
                saveDb.savePlayerStateDao().updateCondition(player.id, (player.condition - conditionLoss).coerceAtLeast(0))
            }
        }

        return events
    }

    private fun calculateTrainingQuality(club: SaveClubStateEntity, ctx: AdvanceContext): Double {
        // 基础设施 × 教练水平 × 球员士气
        val facility = club.trainingLevel / 100.0
        val coach = 0.8  // V1 简化，待 T0M6-2 教练组完善
        val morale = (club.dressingRoomMorale + 50) / 150.0
        return facility * coach * morale
    }
}
```

```kotlin
// daily/InjuryRecoveryTask.kt
class InjuryRecoveryTask(
    private val saveDb: SaveDatabase,
    private val config: ProgressionConfig
) {

    /**
     * V0.1 11 §二.1.4 伤病恢复
     */
    suspend fun execute(ctx: AdvanceContext): List<AdvanceEvent> {
        val events = mutableListOf<AdvanceEvent>()
        val activeClubs = saveDb.saveClubStateDao().getByLeagues(ctx.saveId, ctx.activeLeagueIds)

        for (club in activeClubs) {
            val injuredPlayers = saveDb.saveInjuryDao()
                .getActiveInjuriesByClub(ctx.saveId, club.clubId)

            for (injury in injuredPlayers) {
                // 检查是否到恢复日期
                val expectedReturn = LocalDate.parse(injury.expectedReturnDate)
                if (ctx.nextDate.isAfter(expectedReturn) || ctx.nextDate.isEqual(expectedReturn)) {
                    // 恢复
                    saveDb.saveInjuryDao().updateStatus(injury.injuryId, "recovered")
                    saveDb.savePlayerStateDao().updateInjuryStatus(injury.playerId, "healthy", null)

                    // 医疗水平影响恢复质量
                    val fullRecovery = calculateFullRecoveryChance(club)
                    if (Random.nextDouble() < fullRecovery) {
                        // 完全恢复
                        saveDb.savePlayerStateDao().updateCondition(injury.playerId, 85)
                    } else {
                        // 部分恢复，有复发风险
                        saveDb.savePlayerStateDao().updateCondition(injury.playerId, 70)
                    }

                    events.add(AdvanceEvent(
                        type = AdvanceEventType.INJURY_RECOVERED,
                        description = "球员 ${injury.playerId} 伤病恢复",
                        clubId = club.clubId,
                        playerId = injury.playerId,
                        priority = if (club.clubId == ctx.managerClubId) EventPriority.HIGH else EventPriority.LOW
                    ))
                }
            }
        }

        return events
    }
}
```

```kotlin
// active/ActiveScopeManager.kt
class ActiveScopeManager(
    private val saveDb: SaveDatabase,
    private val historyDb: HistoryDatabase,
    private val config: ProgressionConfig
) {

    /**
     * V0.2 性能优化关键：活跃范围机制
     * - 玩家所在联赛：全部俱乐部 ACTIVE
     * - 玩家俱乐部：FULL
     * - 其他顶级联赛：LIGHT（仅月度/转会窗）
     * - 低级联赛：MINIMAL（仅赛季末）
     */
    fun getActiveLeagueIds(): List<Int> {
        // 玩家所在联赛 + 主要关联联赛（如欧冠参赛国）
        return config.activeLeagues  // 默认 [EPL, SerieA]
    }

    fun getClubDepth(clubId: Int, ctx: AdvanceContext): ClubSimulationDepth {
        return when {
            clubId == ctx.managerClubId -> ClubSimulationDepth.FULL
            isClubInActiveLeagues(clubId, ctx) -> ClubSimulationDepth.ACTIVE
            isClubInTopLeague(clubId) -> ClubSimulationDepth.LIGHT
            else -> ClubSimulationDepth.MINIMAL
        }
    }

    /**
     * 动态扩展活跃范围
     * - 玩家转会到其他联赛 → 新联赛变活跃
     * - 玩家参加欧冠 → 欧冠对手联赛临时活跃
     */
    fun expandActiveScope(newLeagueId: Int) {
        if (newLeagueId !in config.activeLeagues) {
            config.activeLeagues = config.activeLeagues + newLeagueId
        }
    }
}
```

## 九、回滚机制

```kotlin
// rollback/AdvanceRollback.kt
class AdvanceRollback(
    private val saveDb: SaveDatabase,
    private val checkpointManager: CheckpointManager
) {

    /**
     * V0.1 11 §十："如果推进失败，回滚到上一日"
     */
    fun createPreAdvanceSnapshot(): PreAdvanceSnapshot {
        return PreAdvanceSnapshot(
            currentDate = getCurrentDate(),
            saveDbState = captureDbState(),
            checkpointId = checkpointManager.createLightCheckpoint()
        )
    }

    fun rollback(snapshot: PreAdvanceSnapshot) {
        // 1. 关闭当前 db 连接
        saveDb.close()

        // 2. 从 checkpoint 恢复
        checkpointManager.restoreFromCheckpoint(snapshot.checkpointId)

        // 3. 重新打开 db
        saveDb.reopen()

        // 4. 验证日期回滚
        val restoredDate = getCurrentDate()
        if (restoredDate != snapshot.currentDate) {
            throw IllegalStateException("回滚失败：日期未恢复")
        }
    }

    private fun captureDbState(): ByteArray {
        // 捕获关键表状态（用于校验）
        return ByteArray(0)  // 简化
    }
}

data class PreAdvanceSnapshot(
    val currentDate: LocalDate,
    val saveDbState: ByteArray,
    val checkpointId: String
)
```

## 十三、关键风险与应对

### 1. 性能不达标（P95 > 3 秒）

- **表现**：每日推进卡顿，影响体验
- **应对**：
  - 收紧活跃范围（仅玩家联赛 ACTIVE）
  - 非活跃范围改用快速比分生成
  - 异步化新闻/统计任务
  - 数据库批量操作 + 事务
- **红线**：不削减 FULL 范围（玩家俱乐部必须完整模拟）

### 2. 推进失败导致存档损坏

- **表现**：异常中断后存档无法加载
- **应对**：
  - 每次推进前创建 light checkpoint
  - 失败立即回滚到 checkpoint
  - 关键操作事务化
- **验证**：W35 测试人为注入异常，验证回滚有效

### 3. 赛季结束任务失败

- **表现**：升降级/赛程生成异常
- **应对**：
  - 赛季结束前强制 season checkpoint
  - 失败回滚到赛季末前
  - 分阶段提交（每完成 1 项 9 项任务就保存）

### 4. 活跃范围切换异常

- **表现**：玩家转会到其他联赛后，新联赛未变活跃
- **应对**：
  - 转会完成时调用 `expandActiveScope`
  - 欧冠抽签时临时扩展活跃范围
