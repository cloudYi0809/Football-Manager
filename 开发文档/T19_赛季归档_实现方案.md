```kotlin
// summary/SeasonSummaryGenerator.kt
class SeasonSummaryGenerator(
    private val saveDb: SaveDatabase,
    private val historyDb: HistoryDatabase
) {

    /**
     * V0.2 §七.1 生成赛季摘要
     */
    suspend fun generate(ctx: AdvanceContext): SeasonSummary {
        val seasonId = ctx.currentSeasonId

        // 1. 联赛积分榜
        val leagueStandings = generateLeagueStandings(ctx)

        // 2. 射手榜
        val topScorers = generateTopScorers(ctx)

        // 3. 助攻榜
        val topAssists = generateTopAssists(ctx)

        // 4. 转会汇总
        val transfers = generateTransferSummary(ctx)

        // 5. 玩家俱乐部财政
        val managerFinancial = generateManagerFinancial(ctx)

        // 6. 奖项
        val awards = generateAwards(ctx)

        // 7. 升降级
        val (promotions, relegations) = generatePromotionRelegation(ctx)

        return SeasonSummary(
            seasonId = seasonId,
            seasonLabel = "${getSeasonYearStart(seasonId)}/${getSeasonYearEnd(seasonId)}",
            leagueStandings = leagueStandings,
            topScorers = topScorers,
            topAssists = topAssists,
            transfers = transfers,
            managerClubFinancial = managerFinancial,
            awards = awards,
            promotions = promotions,
            relegations = relegations
        )
    }

    private suspend fun generateLeagueStandings(ctx: AdvanceContext): List<LeagueStandingSummary> {
        val activeLeagues = ctx.activeLeagueIds
        return activeLeagues.map { leagueId ->
            val standings = historyDb.matchDao().calculateStandings(ctx.currentSeasonId, leagueId, ctx.currentDate)
            LeagueStandingSummary(
                leagueId = leagueId,
                leagueName = getLeagueName(leagueId),
                standings = standings.map { entity ->
                    StandingEntry(
                        position = entity.position,
                        clubId = entity.clubId,
                        clubName = getClubName(entity.clubId),
                        played = entity.played,
                        won = entity.won,
                        drawn = entity.drawn,
                        lost = entity.lost,
                        goalsFor = entity.goalsFor,
                        goalsAgainst = entity.goalsAgainst,
                        points = entity.points
                    )
                }
            )
        }
    }

    private suspend fun generateTopScorers(ctx: AdvanceContext): List<ScorerListSummary> {
        // 从比赛事件聚合射手榜
        val scorers = saveDb.matchEventDao()
            .getTopScorersBySeason(ctx.saveId, ctx.currentSeasonId, limit = 20)

        return scorers.map { scorer ->
            ScorerListSummary(
                playerId = scorer.playerId,
                playerName = getPlayerName(scorer.playerId),
                clubId = scorer.clubId,
                clubName = getClubName(scorer.clubId),
                goals = scorer.goalCount,
                matches = scorer.matchCount
            )
        }
    }

    private suspend fun generateTransferSummary(ctx: AdvanceContext): TransferSummary {
        val transfers = saveDb.saveTransferOfferDao()
            .getCompletedTransfersBySeason(ctx.saveId, ctx.currentSeasonId)

        return TransferSummary(
            totalTransfers = transfers.size,
            totalFee = transfers.sumOf { it.fee },
            maxFee = transfers.maxOfOrNull { it.fee } ?: 0,
            topTransfers = transfers.sortedByDescending { it.fee }.take(10).map { transfer ->
                TransferRecord(
                    playerId = transfer.playerId,
                    playerName = getPlayerName(transfer.playerId),
                    fromClubId = transfer.fromClubId,
                    fromClubName = getClubName(transfer.fromClubId),
                    toClubId = transfer.toClubId,
                    toClubName = getClubName(transfer.toClubId),
                    fee = transfer.fee,
                    date = transfer.createdDate
                )
            }
        )
    }
}
```

## 四、事件压缩器

```kotlin
// compress/EventCompressor.kt
class EventCompressor(
    private val saveDb: SaveDatabase,
    private val config: ArchiveConfig
) {

    /**
     * V0.2 §七.2 压缩比赛事件
     * 仅保留：比分 + 进球 + 红黄牌 + 关键评分
     * 删除：控球/射门/角球/犯规等详细事件
     */
    suspend fun compress(ctx: AdvanceContext): Int {
        val seasonMatches = saveDb.matchEventDao()
            .getMatchesBySeason(ctx.saveId, ctx.currentSeasonId)

        var compressedCount = 0

        for (match in seasonMatches) {
            // 1. 提取压缩数据
            val compressed = CompressedMatchEvent(
                matchId = match.matchId,
                homeScore = match.homeScore,
                awayScore = match.awayScore,
                homeXg = match.homeXg,
                awayXg = match.awayXg,
                goals = extractGoals(match.events),
                cards = extractCards(match.events),
                topRatedPlayers = extractTopRatings(match.playerRatings, limit = 5)
            )

            // 2. 保存压缩后的事件
            saveDb.compressedMatchDao().insert(compressed)

            // 3. 删除原始详细事件
            saveDb.matchEventDao().deleteByMatch(match.matchId)

            compressedCount++
        }

        return compressedCount
    }

    private fun extractGoals(events: List<MatchEvent>): List<CompressedGoal> {
        return events.filter { it.type == EventType.GOAL }.map { event ->
            CompressedGoal(
                minute = event.minute,
                scorerId = event.playerId,
                assistId = event.secondaryPlayerId
            )
        }
    }

    private fun extractCards(events: List<MatchEvent>): List<CompressedCard> {
        return events.filter { it.type in listOf(EventType.YELLOW_CARD, EventType.RED_CARD) }
            .map { event ->
                CompressedCard(
                    minute = event.minute,
                    playerId = event.playerId!!,
                    cardType = if (event.type == EventType.YELLOW_CARD) "yellow" else "red"
                )
            }
    }

    private fun extractTopRatings(ratings: Map<String, Double>, limit: Int): List<CompressedRating> {
        return ratings.entries
            .sortedByDescending { it.value }
            .take(limit)
            .map { CompressedRating(playerId = it.key.toInt(), rating = it.value) }
    }
}
```

## 五、数据清理器

```kotlin
// cleanup/DataCleaner.kt
class DataCleaner(
    private val saveDb: SaveDatabase,
    private val config: ArchiveConfig
) {

    /**
     * V0.2 §七.3 清理过期数据
     */
    suspend fun clean(ctx: AdvanceContext): Int {
        var cleanedCount = 0

        // 1. 删除过期新闻（保留最近 1 赛季）
        cleanedCount += saveDb.saveNewsDao()
            .deleteNewsBeforeSeason(ctx.saveId, ctx.currentSeasonId - 1)

        // 2. 删除过期待办（已完成 + 过期）
        cleanedCount += saveDb.todoDao()
            .deleteCompletedAndExpired(ctx.saveId, ctx.currentDate.minusMonths(3))

        // 3. 删除过期球探报告（保留最近 6 个月）
        cleanedCount += saveDb.scoutReportDao()
            .deleteReportsBefore(ctx.saveId, ctx.currentDate.minusMonths(6))

        // 4. 删除过期转会报价（已完成 + 过期）
        cleanedCount += saveDb.saveTransferOfferDao()
            .deleteCompletedAndExpired(ctx.saveId, ctx.currentDate.minusMonths(6))

        // 5. 删除过期 AI 决策日志（保留最近 2 赛季）
        cleanedCount += saveDb.aiDecisionLogDao()
            .deleteLogsBeforeSeason(ctx.saveId, ctx.currentSeasonId - 2)

        // 6. 删除过期性能日志（保留最近 1 赛季）
        cleanedCount += saveDb.perfLogDao()
            .deleteLogsBeforeSeason(ctx.saveId, ctx.currentSeasonId - 1)

        return cleanedCount
    }
}
```

## 六、VACUUM 执行器

```kotlin
// vacuum/VacuumExecutor.kt
class VacuumExecutor(private val saveDb: SaveDatabase) {

    /**
     * V0.2 §七.4 VACUUM 压缩
     * SQLite VACUUM 重建数据库文件，回收空间
     */
    suspend fun vacuum(): VacuumResult {
        val sizeBefore = getDbSize()

        val startTime = System.currentTimeMillis()

        // 执行 VACUUM（需要在事务外执行）
        saveDb.openHelper.writableDatabase.execSQL("VACUUM")

        val duration = System.currentTimeMillis() - startTime
        val sizeAfter = getDbSize()

        return VacuumResult(
            sizeBeforeMb = sizeBefore,
            sizeAfterMb = sizeAfter,
            savedMb = sizeBefore - sizeAfter,
            durationMs = duration
        )
    }

    private fun getDbSize(): Double {
        val dbFile = File(saveDb.openHelper.writableDatabase.path!!)
        return dbFile.length() / (1024.0 * 1024.0)
    }
}

data class VacuumResult(
    val sizeBeforeMb: Double,
    val sizeAfterMb: Double,
    val savedMb: Double,
    val durationMs: Long
)
```

## 七、入口服务

```kotlin
// api/SeasonArchiver.kt
class SeasonArchiver(
    private val summaryGenerator: SeasonSummaryGenerator,
    private val eventCompressor: EventCompressor,
    private val statsAggregator: StatsAggregator,
    private val rankingFreezer: RankingFreezer,
    private val dataCleaner: DataCleaner,
    private val vacuumExecutor: VacuumExecutor,
    private val saveDb: SaveDatabase,
    private val config: ArchiveConfig
) {

    /**
     * V0.2 §七 完整归档流程
     * 由 T07 SeasonEndExecutor 调用
     */
    suspend fun archiveSeason(ctx: AdvanceContext): ArchiveResult {
        val startTime = System.currentTimeMillis()
        val sizeBefore = getDbSize()

        // 1. 生成赛季摘要
        val summary = summaryGenerator.generate(ctx)

        // 2. 压缩比赛事件
        val compressedMatches = eventCompressor.compress(ctx)

        // 3. 聚合统计
        statsAggregator.aggregate(ctx)

        // 4. 冻结榜单
        rankingFreezer.freeze(ctx)

        // 5. 清理过期数据
        val cleanedRecords = dataCleaner.clean(ctx)

        // 6. VACUUM 压缩
        val vacuumResult = vacuumExecutor.vacuum()

        // 7. 保存归档记录
        val archiveId = saveDb.seasonArchiveDao().insert(SeasonArchiveEntity(
            archiveId = 0,
            saveId = ctx.saveId.toString(),
            seasonId = ctx.currentSeasonId,
            archiveType = "full",
            summaryJson = serializeSummary(summary),
            createdAt = ctx.currentDate.toString()
        )).toInt()

        val duration = System.currentTimeMillis() - startTime
        val sizeAfter = getDbSize()

        return ArchiveResult(
            seasonId = ctx.currentSeasonId,
            archiveId = archiveId,
            summary = summary,
            compressedMatches = compressedMatches,
            cleanedRecords = cleanedRecords,
            sizeBeforeMb = sizeBefore,
            sizeAfterMb = sizeAfter,
            durationMs = duration
        )
    }

    /**
     * V0.2 §七 体积监控
     * 每次归档后记录体积
     */
    suspend fun checkSize(ctx: AdvanceContext): SizeReport {
        val currentSize = getDbSize()
        val seasonsArchived = saveDb.seasonArchiveDao().countBySave(ctx.saveId.toString())

        return SizeReport(
            currentSizeMb = currentSize,
            seasonsArchived = seasonsArchived,
            maxSizeMb = config.maxSaveSizeMb,  // 默认 80MB
            isApproachingLimit = currentSize > config.maxSaveSizeMb * 0.85,
            isOverLimit = currentSize > config.maxSaveSizeMb
        )
    }

    private fun getDbSize(): Double {
        val dbFile = File(saveDb.openHelper.writableDatabase.path!!)
        return dbFile.length() / (1024.0 * 1024.0)
    }
}

data class SizeReport(
    val currentSizeMb: Double,
    val seasonsArchived: Int,
    val maxSizeMb: Double,
    val isApproachingLimit: Boolean,
    val isOverLimit: Boolean
)
```

```kotlin
class HistoricalSeasonQueryService(
    private val saveDb: SaveDatabase
) {

    /**
     * V0.2 §七.5 旧赛季查询接口
     * 历史赛季只读访问
     */
    suspend fun getSeasonArchive(saveId: String, seasonId: Int): SeasonArchiveEntity? {
        return saveDb.seasonArchiveDao().getBySeason(saveId, seasonId)
    }

    suspend fun getSeasonSummary(saveId: String, seasonId: Int): SeasonSummary? {
        val archive = getSeasonArchive(saveId, seasonId) ?: return null
        return deserializeSummary(archive.summaryJson)
    }

    suspend fun getHistoricalStandings(saveId: String, seasonId: Int, leagueId: String): List<StandingEntry>? {
        val summary = getSeasonSummary(saveId, seasonId) ?: return null
        return summary.leagueStandings.find { it.leagueId == leagueId }?.standings
    }

    suspend fun getHistoricalTopScorers(saveId: String, seasonId: Int): List<ScorerListSummary>? {
        val summary = getSeasonSummary(saveId, seasonId) ?: return null
        return summary.topScorers
    }

    suspend fun getAllArchivedSeasons(saveId: String): List<SeasonArchiveEntity> {
        return saveDb.seasonArchiveDao().getAllBySave(saveId)
    }

    /**
     * 查询压缩后的比赛结果（只读）
     */
    suspend fun getCompressedMatch(matchId: Int): CompressedMatchEvent? {
        return saveDb.compressedMatchDao().getByMatch(matchId)
    }
}
```

## 十一、关键风险与应对

### 1. 体积超标（>80MB）

- **表现**：20 年存档体积超过 80MB
- **应对**：
  - 加严清理策略（news_keep_seasons=0）
  - 压缩更多事件（删除 Top5 评分）
  - 定期 VACUUM
- **红线**：20 年必须 ≤80MB

### 2. 归档耗时过长

- **表现**：归档超过 10 秒
- **应对**：
  - 异步归档（后台执行）
  - 分批压缩（每次 100 场）
  - 显示归档进度

### 3. 历史查询卡顿

- **表现**：查询旧赛季积分榜慢
- **应对**：
  - 摘要预生成 JSON
  - 榜单冻结到 ranking_cache
  - 索引优化
