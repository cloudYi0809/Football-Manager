## 三、赛程生成算法

### 1. 联赛双循环（Circle Method 圆盘旋转法）

```kotlin
// generator/LeagueScheduleGenerator.kt
class LeagueScheduleGenerator(
    private val config: ScheduleConfig,
    private val random: Random = Random(System.currentTimeMillis())
) {

    /**
     * V0.1 11 §九.2 双循环联赛
     * - 主客场平衡
     * - 避免同队连续过多主场或客场（≤3 连主/连客）
     *
     * @param clubs 参赛俱乐部（数量须为偶数，奇数时补 bye）
     * @param seasonStart 赛季开始日期
     * @return 完整赛程（2*(n-1) 轮）
     */
    fun generateDoubleRoundRobin(
        clubs: List<Int>,
        seasonStart: LocalDate,
        competitionId: Int,
        seasonId: Int
    ): List<SaveMatchEntity> {
        // 1. 球队数处理：奇数补虚拟队（bye），轮空方不安排比赛
        val teams = clubs.toMutableList()
        val hasBye = teams.size % 2 != 0
        if (hasBye) teams.add(BYE_TEAM_ID)

        val n = teams.size
        val roundsPerHalf = n - 1            // 半程轮次数
        val totalRounds = 2 * roundsPerHalf   // 双循环总轮次
        val matchesPerRound = n / 2

        // 2. 圆盘旋转法生成首轮对阵
        val firstHalfMatches = mutableListOf<MutableList<MatchPair>>()
        val rotation = teams.toMutableList()

        for (round in 0 until roundsPerHalf) {
            val roundPairs = mutableListOf<MatchPair>()
            for (i in 0 until matchesPerRound) {
                val home = rotation[i]
                val away = rotation[n - 1 - i]
                if (home != BYE_TEAM_ID && away != BYE_TEAM_ID) {
                    // 交替主客：偶数轮 home 在前，奇数轮对换，保证主客平衡
                    val pair = if (round % 2 == 0) {
                        MatchPair(home, away)
                    } else {
                        MatchPair(away, home)
                    }
                    roundPairs.add(pair)
                }
            }
            firstHalfMatches.add(roundPairs)

            // 旋转：固定第一位，其余左移一位
            val fixed = rotation[0]
            rotation.removeAt(0)
            rotation.add(0, fixed)
            // 最后一位移到第二位
            val last = rotation.removeAt(rotation.size - 1)
            rotation.add(1, last)
        }

        // 3. 第二半程：主客对换
        val secondHalfMatches = firstHalfMatches.map { round ->
            round.map { MatchPair(it.away, it.home) }.toMutableList()
        }

        // 4. 随机化轮次顺序（避免可预测），但保持主客半程对应
        val firstHalfOrder = (0 until roundsPerHalf).toMutableList().also { random.shuffle(it) }
        val secondHalfOrder = (0 until roundsPerHalf).toMutableList().also { random.shuffle(it) }

        // 5. 连续主客检查与调整（≤3 连主/连客）
        val finalSchedule = mutableListOf<List<MatchPair>>()
        firstHalfOrder.forEachIndexed { idx, origRound ->
            finalSchedule.add(firstHalfMatches[origRound])
        }
        secondHalfOrder.forEachIndexed { idx, origRound ->
            finalSchedule.add(secondHalfMatches[origRound])
        }
        adjustConsecutiveHomeAway(finalSchedule, clubs)

        // 6. 分配日期（按轮次间隔 + 杯赛/欧战/国际比赛日避让）
        val matchDates = assignMatchDates(
            totalRounds, seasonStart, config.leagueRoundIntervalDays
        )

        // 7. 转为 SaveMatchEntity
        val result = mutableListOf<SaveMatchEntity>()
        finalSchedule.forEachIndexed { roundIndex, roundPairs ->
            roundPairs.forEach { pair ->
                result.add(SaveMatchEntity(
                    saveId = 0, // 由调用方填入
                    seasonId = seasonId,
                    competitionId = competitionId,
                    round = roundIndex + 1,
                    matchDate = matchDates[roundIndex].toString(),
                    homeClubId = pair.home,
                    awayClubId = pair.away,
                    stage = "regular",
                    leg = 1
                ))
            }
        }
        return result
    }

    /**
     * 连续主客场调整：保证任意球队不连续超过 3 场主场或客场
     */
    private fun adjustConsecutiveHomeAway(
        schedule: MutableList<List<MatchPair>>,
        clubs: List<Int>
    ) {
        val maxConsecutive = config.maxConsecutiveHomeAway  // 默认 3
        for (attempt in 0 until config.adjustMaxAttempts) { // 默认 50 次
            var violations = 0
            for (club in clubs) {
                val sequence = schedule.map { round ->
                    round.find { it.home == club || it.away == club }
                        ?.let { if (it.home == club) 'H' else 'A' } ?: '-'
                }.joinToString("")
                if (hasLongRun(sequence, 'H', maxConsecutive) ||
                    hasLongRun(sequence, 'A', maxConsecutive)) {
                    violations++
                }
            }
            if (violations == 0) return
            // 随机交换两轮的内部主客对换，尝试减少违规
            swapRandomRoundHomeAway(schedule, random)
        }
        // 超过最大尝试仍违规：记录警告，不阻塞生成
    }

    private fun hasLongRun(seq: String, ch: Char, maxLen: Int): Boolean {
        var run = 0
        for (c in seq) {
            run = if (c == ch) run + 1 else 0
            if (run > maxLen) return true
        }
        return false
    }

    data class MatchPair(val home: Int, val away: Int)

    companion object {
        const val BYE_TEAM_ID = -1
    }
}
```

### 2. 杯赛对阵表生成（单/双回合淘汰）

```kotlin
// generator/CupBracketGenerator.kt
class CupBracketGenerator(private val config: ScheduleConfig) {

    /**
     * V0.1 11 §九 杯赛赛程
     * - 单回合淘汰 / 双回合淘汰
     * - 种子排位避免强强过早相遇
     *
     * @param participants 参赛俱乐部（须为 2 的幂，不足则轮空 bye）
     * @param seedRanking 种子排名（俱乐部Id -> 种子分），用于排位
     * @param stageConfig 各轮次赛制（如 32强单回合、半决赛双回合）
     */
    fun generateCupBracket(
        participants: List<Int>,
        seedRanking: Map<Int, Int>,
        stageConfig: List<CupStageConfig>,
        competitionId: Int,
        seasonId: Int,
        startDate: LocalDate
    ): Pair<List<SaveCupTieEntity>, List<SaveMatchEntity>> {
        // 1. 补齐到 2 的幂（轮空处理）
        val targetSize = nextPowerOfTwo(participants.size)
        val slots = participants.sortedBy { seedRanking[it] ?: Int.MAX_VALUE }.toMutableList()
        while (slots.size < targetSize) slots.add(null)  // null = 轮空

        // 2. 种子对阵排位（标准对阵表 seeding）
        // 1 vs 16, 8 vs 9, 5 vs 12, 4 vs 13, ... 保证强种子分散
        val seededOrder = seedBracketSlots(targetSize)
        val firstRoundPairs = mutableListOf<Pair<Int?, Int?>>()
        for (i in seededOrder.indices step 2) {
            firstRoundPairs.add(slots[seededOrder[i]] to slots[seededOrder[i + 1]])
        }

        // 3. 生成首轮 ties
        val ties = mutableListOf<SaveCupTieEntity>()
        val matches = mutableListOf<SaveMatchEntity>()
        var currentDate = startDate
        val firstStage = stageConfig.first()

        firstRoundPairs.forEachIndexed { idx, (home, away) ->
            val tieId = "${competitionId}_${seasonId}_R${firstStage.stageOrder}_T${idx}"
            // 轮空直接晋级
            val winner = when {
                home == null && away != null -> away
                away == null && home != null -> home
                home == null && away == null -> null
                else -> null  // 待比赛决出
            }
            ties.add(SaveCupTieEntity(
                tieId = tieId,
                saveId = 0,
                seasonId = seasonId,
                competitionId = competitionId,
                stage = firstStage.stage,
                stageOrder = firstStage.stageOrder,
                homeClubId = home,
                awayClubId = away,
                firstLegMatchId = null,
                secondLegMatchId = null,
                winnerClubId = winner,
                isTwoLegged = if (firstStage.isTwoLegged) 1 else 0,
                slotIndex = idx,
                nextTieId = computeNextTieId(competitionId, seasonId, firstStage.stageOrder, idx)
            ))
            // 轮空不生成比赛
            if (home != null && away != null && winner == null) {
                matches.addAll(createTieMatches(
                    tieId, home, away, firstStage, currentDate, competitionId, seasonId
                ))
            }
        }

        // 4. 后续轮次 ties 预创建（参赛方待定，由晋级回填）
        var prevRoundCount = firstRoundPairs.size
        var prevStage = firstStage
        for (stageIndex in 1 until stageConfig.size) {
            val stage = stageConfig[stageIndex]
            val tieCount = prevRoundCount / 2
            currentDate = currentDate.plusDays(stage.intervalDays.toLong())
            for (i in 0 until tieCount) {
                ties.add(SaveCupTieEntity(
                    tieId = "${competitionId}_${seasonId}_R${stage.stageOrder}_T${i}",
                    saveId = 0,
                    seasonId = seasonId,
                    competitionId = competitionId,
                    stage = stage.stage,
                    stageOrder = stage.stageOrder,
                    homeClubId = null,    // 待回填
                    awayClubId = null,
                    isTwoLegged = if (stage.isTwoLegged) 1 else 0,
                    slotIndex = i,
                    nextTieId = computeNextTieId(
                        competitionId, seasonId, stage.stageOrder, i
                    )
                ))
            }
            prevRoundCount = tieCount
            prevStage = stage
        }

        return ties to matches
    }

    /**
     * 标准对阵表种子位（保证 1、2 种子分属两个半区）
     * 8 队：[0,7,3,4,5,2,6,1] → 1v8,4v5,3v6,2v7
     */
    private fun seedBracketSlots(size: Int): List<Int> {
        val result = mutableListOf(0, size - 1)
        while (result.size < size) {
            val newLayer = mutableListOf<Int>()
            for (slot in result) {
                newLayer.add(slot)
                newLayer.add(size - 1 - slot)
            }
            result.clear()
            result.addAll(newLayer.distinct())
        }
        return result.subList(0, size)
    }

    private fun nextPowerOfTwo(n: Int): Int {
        var p = 2
        while (p < n) p *= 2
        return p
    }

    private fun createTieMatches(
        tieId: String, home: Int, away: Int,
        stage: CupStageConfig, date: LocalDate,
        competitionId: Int, seasonId: Int
    ): List<SaveMatchEntity> {
        return if (stage.isTwoLegged) {
            listOf(
                SaveMatchEntity(saveId = 0, seasonId = seasonId, competitionId = competitionId,
                    round = stage.stageOrder, matchDate = date.toString(),
                    homeClubId = home, awayClubId = away, stage = stage.stage, leg = 1, tieId = tieId),
                SaveMatchEntity(saveId = 0, seasonId = seasonId, competitionId = competitionId,
                    round = stage.stageOrder,
                    matchDate = date.plusDays(7).toString(),
                    homeClubId = away, awayClubId = home, stage = stage.stage, leg = 2, tieId = tieId)
            )

            listOf(SaveMatchEntity(saveId = 0, seasonId = seasonId, competitionId = competitionId,
                round = stage.stageOrder, matchDate = date.toString(),
```

```kotlin
// generator/EuroScheduleGenerator.kt
class EuroScheduleGenerator(
    private val leagueGenerator: LeagueScheduleGenerator,
    private val cupGenerator: CupBracketGenerator,
    private val config: ScheduleConfig
) {

    /**
     * V0.1 02 §26 欧战
     * - 小组赛：8 组 4 队双循环（6 轮）
     * - 淘汰赛：16 强起单/双回合淘汰
     */
    fun generateEuroCompetition(
        participants: List<Int>,
        seedRanking: Map<Int, Int>,
        competitionId: Int,
        seasonId: Int,
        groupStageStart: LocalDate,
        knockoutStart: LocalDate
    ): Pair<List<SaveCupTieEntity>, List<SaveMatchEntity>> {
        val allTies = mutableListOf<SaveCupTieEntity>()
        val allMatches = mutableListOf<SaveMatchEntity>()

        // 1. 分组：4 档抽签，每组 4 队，避免同足协
        val groups = drawGroups(participants, seedRanking, config.euroGroupCount)

        // 2. 小组赛：每组双循环 6 轮
        groups.forEach { group ->
            val groupMatches = leagueGenerator.generateDoubleRoundRobin(
                group.clubs, groupStageStart, competitionId, seasonId
            ).map { it.copy(stage = "group", groupId = group.id) }
            allMatches.addAll(groupMatches)
        }

        // 3. 小组出线：每组前 2 名晋级 16 强（16 队）
        //    出线名单在小组赛结束后由 LeagueTableCalculator 计算
        //    此处预创建 16 强淘汰 ties（参赛方待回填）
        val knockoutStages = listOf(
            CupStageConfig("round_of_16", 1, isTwoLegged = true, intervalDays = 14),
            CupStageConfig("quarter", 2, isTwoLegged = true, intervalDays = 14),
            CupStageConfig("semi", 3, isTwoLegged = true, intervalDays = 14),
            CupStageConfig("final", 4, isTwoLegged = false, intervalDays = 21)
        )
        val (ties, matches) = cupGenerator.generateCupBracket(
            participants = (1..16).toList(),  // 占位，实际出线后回填
            seedRanking = emptyMap(),
            stageConfig = knockoutStages,
            competitionId = competitionId,
            seasonId = seasonId,
            startDate = knockoutStart
        )
        allTies.addAll(ties)
        allMatches.addAll(matches)

        return allTies to allMatches
    }

    private fun drawGroups(
        participants: List<Int>,
        seedRanking: Map<Int, Int>,
        groupCount: Int
    ): List<EuroGroup> {
        // 按 seedRanking 分 4 档
        val sorted = participants.sortedBy { seedRanking[it] ?: Int.MAX_VALUE }
        val potSize = groupCount  // 每档 groupCount 队
        val pots = sorted.chunked(potSize)
        // 从每档抽一队入每组（V1 简化：不强制同足协回避，可后期加约束）
        val groups = (0 until groupCount).map { idx ->
            EuroGroup(
                id = (('A' + idx / 26).toString() + (idx % 26 + 1)),
                clubs = pots.map { pot -> pot.getOrElse(idx) { -1 } }.filter { it != -1 }
            )
        }
        return groups
    }
}
```

```kotlin
class NationalScheduleGenerator(

     * V0.1 11 §七 国家队 AI
     * - 国际比赛日窗口征召球员
     * - 窗口内安排 1-2 场国家队比赛
     * - 球员增加疲劳，可能受伤
     */
    fun generateNationalWindows(
        seasonId: Int,
        seasonStart: LocalDate,
        seasonEnd: LocalDate
    ): List<NationalWindow> {
        val windows = mutableListOf<NationalWindow>()
        var date = seasonStart
        while (date.isBefore(seasonEnd)) {
            // 配置的国际比赛日日期（如 9/10/11/3/6 月）
            val matchedWindow = config.internationalWindows.find { window ->
                window.month == date.monthValue
            }
            if (matchedWindow != null) {
                val windowStart = LocalDate.of(date.year, matchedWindow.month, matchedWindow.startDay)
                windows.add(NationalWindow(
                    windowStart = windowStart,
                    windowEnd = windowStart.plusDays(matchedWindow.durationDays.toLong()),
                    matchCount = matchedWindow.matchCount
                ))
            }
            date = date.plusMonths(1)
        }
        return windows
    }

    /**
     * 国际比赛日征召：从各俱乐部抽调国脚
     */
    suspend fun processCallups(
        saveId: Int,
        window: NationalWindow,
        currentDate: LocalDate
    ): List<SaveInternationalCallupEntity> {
        val callups = mutableListOf<SaveInternationalCallupEntity>()
        // 1. 查询所有可被征召球员（有 nationality 且能力达国家队门槛）

            saveId, minCa = config.nationalTeamMinCa
        )
        // 2. 按国籍分组，每国征召 23 人
        val byNation = eligible.groupBy { it.nationality }
        byNation.forEach { (nation, players) ->
            val squad = players.sortedByDescending { it.currentCa }.take(23)

                    windowEnd = window.windowEnd.toString(),
                    matchCount = window.matchCount,
                    isReleased = 1,
                    fatigueAdded = window.matchCount * config.callupFatiguePerMatch
                ))
            }
        }
        return callups
    }
```

### 5. 赛程编排（穿插避冲突）

```kotlin
// generator/ScheduleOrchestrator.kt
class ScheduleOrchestrator(
    private val leagueGen: LeagueScheduleGenerator,
    private val cupGen: CupBracketGenerator,
    private val euroGen: EuroScheduleGenerator,
    private val nationalGen: NationalScheduleGenerator,
    private val config: ScheduleConfig
) {

    /**
     * 赛季开始时编排完整赛程
     * 优先级：国际比赛日 > 联赛 > 杯赛 > 欧战
     * 黑名单日（国际比赛日窗口）不排俱乐部比赛
     */
    suspend fun orchestrateSeason(
        saveId: Int,
        seasonId: Int,
        seasonStart: LocalDate,
        seasonEnd: LocalDate,
        leagueParticipants: Map<Int, List<Int>>,   // competitionId -> clubs
        cupParticipants: Map<Int, List<Int>>,
        euroParticipants: Map<Int, List<Int>>
    ): OrchestrationResult {

        // 1. 计算国际比赛日黑名单
        val blackoutDates = nationalGen.generateNationalWindows(seasonId, seasonStart, seasonEnd)
            .flatMap { window ->
                (window.windowStart..window.windowEnd).toList()
            }.toSet()

        // 2. 生成联赛赛程（占据周中/周末主干）
        val leagueMatches = mutableListOf<SaveMatchEntity>()
        leagueParticipants.forEach { (compId, clubs) ->
            val matches = leagueGen.generateDoubleRoundRobin(
                clubs, seasonStart, compId, seasonId
            ).map { it.copy(saveId = saveId) }
            // 重新分配日期避开黑名单
            val adjusted = redistributeDates(matches, blackoutDates, config.leagueRoundIntervalDays)
            leagueMatches.addAll(adjusted)
        }

        // 3. 生成杯赛赛程（占用联赛空档周）
        val cupTies = mutableListOf<SaveCupTieEntity>()
        val cupMatches = mutableListOf<SaveMatchEntity>()
        cupParticipants.forEach { (compId, participants) ->
            val cupStart = findNextAvailableDate(
                seasonStart, blackoutDates, leagueMatches.map { it.matchDate }.toSet()
            )
            val (ties, matches) = cupGen.generateCupBracket(
                participants, emptyMap(),
                config.defaultCupStages, compId, seasonId, cupStart
            )
            cupTies.addAll(ties.map { it.copy(saveId = saveId) })
            cupMatches.addAll(matches.map { it.copy(saveId = saveId) })
        }

        // 4. 生成欧战赛程（占用特定欧战周）
        val euroTies = mutableListOf<SaveCupTieEntity>()
        val euroMatches = mutableListOf<SaveMatchEntity>()
        euroParticipants.forEach { (compId, participants) ->
            val euroGroupStart = seasonStart.plusWeeks(config.euroGroupStartWeekOffset.toLong())
            val euroKnockoutStart = seasonStart.plusMonths(config.euroKnockoutStartMonthOffset.toLong())
            val (ties, matches) = euroGen.generateEuroCompetition(
                participants, emptyMap(), compId, seasonId,
                euroGroupStart, euroKnockoutStart
            )
            euroTies.addAll(ties.map { it.copy(saveId = saveId) })
            euroMatches.addAll(matches.map { it.copy(saveId = saveId) })
        }

        // 5. 合并所有比赛日期，检测冲突
        val allMatches = leagueMatches + cupMatches + euroMatches
        val conflicts = detectDateConflicts(allMatches, config.maxMatchesPerClubPerDay)
        if (conflicts.isNotEmpty()) {
            // 移动冲突比赛到下一个可用日期
            resolveConflicts(conflicts, allMatches, blackoutDates)
        }

        // 6. 标记玩家俱乐部比赛
        val playerClubId = getPlayerClubId(saveId)
        allMatches.forEach { m ->
            if (m.homeClubId == playerClubId || m.awayClubId == playerClubId) {
                m.isPlayerMatch = 1
            }
        }

        return OrchestrationResult(
            leagueMatches = leagueMatches,
            cupTies = cupTies, cupMatches = cupMatches,
            euroTies = euroTies, euroMatches = euroMatches,
            blackoutDates = blackoutDates
        )
    }

    private fun redistributeDates(
        matches: List<SaveMatchEntity>,
        blackoutDates: Set<LocalDate>,
        intervalDays: Int
    ): List<SaveMatchEntity> {
        // 按轮次分组，每轮分配一个日期，跳过黑名单
        val byRound = matches.groupBy { it.round }
        var currentDate = matches.first().matchDate.let { LocalDate.parse(it) }
        return byRound.flatMap { (round, roundMatches) ->
            while (currentDate in blackoutDates) {
                currentDate = currentDate.plusDays(1)
            }
            val result = roundMatches.map { it.copy(matchDate = currentDate.toString()) }
            currentDate = currentDate.plusDays(intervalDays.toLong())
            result
        }
    }
}

data class OrchestrationResult(
    val leagueMatches: List<SaveMatchEntity>,
    val cupTies: List<SaveCupTieEntity>,
    val cupMatches: List<SaveMatchEntity>,
    val euroTies: List<SaveCupTieEntity>,
    val euroMatches: List<SaveMatchEntity>,
    val blackoutDates: Set<LocalDate>
)
```

```kotlin
// table/LeagueTableCalculator.kt
class LeagueTableCalculator(
    private val config: ScheduleConfig,
    private val comparator: StandingComparator
) {

    /**
     * V0.1 11 §九 + T06 任务：胜3平1负0
     * 同分排名规则：积分 > 净胜球 > 进球 > 相互战绩 > 客场进球
     */
    suspend fun recalculate(
        saveId: Int,
        seasonId: Int,
        competitionId: Int,
        finishedMatches: List<SaveMatchEntity>,
        existingTable: List<SaveLeagueTableEntity>
    ): List<SaveLeagueTableEntity> {
        // 1. 累计每队战绩
        val stats = mutableMapOf<Int, TeamStats>()
        existingTable.forEach { entry ->
            stats[entry.clubId] = TeamStats(clubId = entry.clubId)
        }

        for (match in finishedMatches.filter { it.status == "finished" }) {
            val home = stats.getOrPut(match.homeClubId) { TeamStats(match.homeClubId) }
            val away = stats.getOrPut(match.awayClubId) { TeamStats(match.awayClubId) }
            val homeScore = match.homeScoreSim ?: continue
            val awayScore = match.awayScoreSim ?: continue

            home.played++; away.played++
            home.goalsFor += homeScore; home.goalsAgainst += awayScore
            away.goalsFor += awayScore; away.goalsAgainst += homeScore
            when {
                homeScore > awayScore -> {
                    home.won++; home.points += 3
                    away.lost++
                    home.homeWon++
                }
                homeScore < awayScore -> {
                    away.won++; away.points += 3
                    home.lost++
                    away.awayWon++
                }
                else -> {
                    home.drawn++; home.points += 1
                    away.drawn++; away.points += 1
                    home.homeDrawn++; away.awayDrawn++
                }
            }
        }

        // 2. 计算净胜球与近 5 场 form
        stats.values.forEach { s ->
            s.goalDifference = s.goalsFor - s.goalsAgainst
        }
        val formMap = computeRecentForm(finishedMatches, windowSize = 5)

        // 3. 排序
        val sorted = stats.values.sortedWith(comparator)

        // 4. 生成实体
        return sorted.mapIndexed { index, s ->
            SaveLeagueTableEntity(
                saveId = saveId,
                seasonId = seasonId,
                competitionId = competitionId,
                clubId = s.clubId,
                played = s.played, won = s.won, drawn = s.drawn, lost = s.lost,
                goalsFor = s.goalsFor, goalsAgainst = s.goalsAgainst,
                goalDifference = s.goalDifference, points = s.points,
                rank = index + 1,
                homePlayed = s.homeWon + s.homeDrawn + s.homeLost,
                homeWon = s.homeWon, homeDrawn = s.homeDrawn, homeLost = s.homeLost,
                awayPlayed = s.awayWon + s.awayDrawn + s.awayLost,
                awayWon = s.awayWon, awayDrawn = s.awayDrawn, awayLost = s.awayLost,
                formJson = formMap[s.clubId]?.joinToString(""),
                updatedAt = LocalDate.now().toString()
            )
        }
    }
}

data class TeamStats(
    val clubId: Int,
    var played: Int = 0, var won: Int = 0, var drawn: Int = 0, var lost: Int = 0,
    var goalsFor: Int = 0, var goalsAgainst: Int = 0, var goalDifference: Int = 0,
    var points: Int = 0,
    var homeWon: Int = 0, var homeDrawn: Int = 0, var homeLost: Int = 0,
    var awayWon: Int = 0, var awayDrawn: Int = 0, var awayLost: Int = 0
)

// table/StandingComparator.kt
class StandingComparator(
    private val finishedMatches: List<SaveMatchEntity> = emptyList()
) : Comparator<TeamStats> {

    override fun compare(a: TeamStats, b: TeamStats): Int {
        // 1. 积分
        if (a.points != b.points) return b.points.compareTo(a.points)
        // 2. 净胜球
        if (a.goalDifference != b.goalDifference)
            return b.goalDifference.compareTo(a.goalDifference)
        // 3. 进球数
        if (a.goalsFor != b.goalsFor) return b.goalsFor.compareTo(a.goalsFor)
        // 4. 相互交锋战绩
        val h2h = compareHeadToHead(a.clubId, b.clubId, finishedMatches)
        if (h2h != 0) return h2h
        // 5. 客场进球数（可配置开关）
        if (config.useAwayGoalsTiebreaker) {
            if (a.goalsFor - a.homeWon != b.goalsFor - b.homeWon) {
                return (b.goalsFor - b.homeWon).compareTo(a.goalsFor - a.homeWon)
            }
        }
        // 6. 抽签（稳定排序，保持原顺序）
        return 0
    }

    private fun compareHeadToHead(
        clubA: Int, clubB: Int,
        matches: List<SaveMatchEntity>
    ): Int {
        var aPoints = 0; var bPoints = 0
        var aGd = 0; var bGd = 0
        for (m in matches) {
            if (m.status != "finished") continue
            val isAHome = m.homeClubId == clubA && m.awayClubId == clubB
            val isAAway = m.awayClubId == clubA && m.homeClubId == clubB
            if (!isAHome && !isAAway) continue
            val aScore = if (isAHome) m.homeScoreSim else m.awayScoreSim ?: continue
            val bScore = if (isAHome) m.awayScoreSim else m.homeScoreSim ?: continue
            when {
                aScore > bScore -> aPoints += 3
                aScore < bScore -> bPoints += 3
                else -> { aPoints++; bPoints++ }
            }
            aGd += aScore - bScore
            bGd += bScore - aScore
        }
        if (aPoints != bPoints) return bPoints.compareTo(aPoints)
        return bGd.compareTo(aGd)
    }
}
```

```kotlin
     * 性能要求：≤500ms
     */
    suspend fun generateSeasonSchedule(

    ): ScheduleGenerationResult {
        val startMs = System.currentTimeMillis()
        val season = historyDb.seasonDao().getById(seasonId)
            ?: return ScheduleGenerationResult(success = false, message = "赛季不存在")
        val seasonStart = LocalDate.parse(season.startDate)

        // 3. 批量写入数据库（事务）
        saveDb.runInTransaction {
            matchDao.insertAll(result.leagueMatches + result.cupMatches + result.euroMatches)
            cupTieDao.insertAll(result.cupTies + result.euroTies)
            scheduleStateDao.insert(SaveScheduleStateEntity(

                blackoutDatesJson = serializeBlackout(result.blackoutDates)
            ))
        }

        // 4. 初始化积分榜（空表）
        leagueParticipants.forEach { (compId, clubs) ->
            val emptyTable = clubs.map { clubId ->
                SaveLeagueTableEntity(saveId = saveId, seasonId = seasonId,
                    competitionId = compId, clubId = clubId)
            }
            tableDao.insertAll(emptyTable)
        }

        val durationMs = System.currentTimeMillis() - startMs
        perfLog("schedule_generate", durationMs)
        return ScheduleGenerationResult(
            success = true,
            matchCount = result.leagueMatches.size + result.cupMatches.size + result.euroMatches.size,
            durationMs = durationMs,
            message = "赛程生成完成，共 ${result.leagueMatches.size + result.cupMatches.size + result.euroMatches.size} 场"
        )
    }

    /**
     * 比赛结束后更新积分榜（由 T07 比赛日处理调用）
     * 性能要求：≤100ms
     */

        seasonId: Int,

    ): List<SaveLeagueTableEntity> {

        val finishedMatches = matchDao.getFinishedByCompetition(saveId, seasonId, competitionId)
        val existing = tableDao.getByCompetition(saveId, seasonId, competitionId)
        val updated = tableCalculator.recalculate(saveId, seasonId, competitionId, finishedMatches, existing)
        tableDao.updateAll(updated)

        cacheDb.leagueTableCacheDao().upsert(
            LeagueTableCacheEntity(
                saveId = saveId, seasonId = seasonId, competitionId = competitionId,
                rank = 0, clubId = 0, points = 0,
                snapshotJson = serializeTable(updated),
                updatedAt = LocalDate.now().toString()
            )
        )
        val durationMs = System.currentTimeMillis() - startMs
        perfLog("table_update", durationMs)

    }

    /**
     * 杯赛/欧战比赛结束后回填晋级方
     */
    suspend fun resolveCupTie(saveId: Int, tieId: String) {
        val tie = cupTieDao.getById(tieId) ?: return
        val matches = matchDao.getByTie(tieId)
        val winner = computeTieWinner(tie, matches)

                } else {
                    cupTieDao.updateAway(nextId, winner)
                }
                // 若两方都已确定且未生成比赛，则生成比赛
                if (nextTie.homeClubId != null && nextTie.awayClubId != null &&
                    nextTie.firstLegMatchId == null) {

                }

     */
    suspend fun getCupBracket(saveId: Int, competitionId: Int): List<SaveCupTieEntity> {
        return cupTieDao.getByCompetition(saveId, competitionId)
    }

    private fun computeTieWinner(
        tie: SaveCupTieEntity, matches: List<SaveMatchEntity>
    ): Int {
        if (matches.isEmpty()) return -1
        if (!tie.isTwoLegged.toBoolean()) {
            val m = matches.first()
            val home = m.homeScoreSim ?: return -1
            val away = m.awayScoreSim ?: return -1
            return when {
                home > away -> m.homeClubId
                away > home -> m.awayClubId
                else -> {
                    // 平局：加时/点球（V1 简化为随机，可扩展）
                    if (config.cupUsePenaltyShootout) {
                        if (Random.nextBoolean()) m.homeClubId else m.awayClubId
                    } else -1
                }
            }
        }
        // 双回合：总比分
        val leg1 = matches.find { it.leg == 1 }
        val leg2 = matches.find { it.leg == 2 }
        val homeAgg = (leg1?.homeScoreSim ?: 0) + (leg2?.awayScoreSim ?: 0)
        val awayAgg = (leg1?.awayScoreSim ?: 0) + (leg2?.homeScoreSim ?: 0)
        return when {
            homeAgg > awayAgg -> tie.homeClubId!!
            awayAgg > homeAgg -> tie.awayClubId!!
            else -> {
                // 客场进球规则
                val awayGoalsA = (leg1?.awayScoreSim ?: 0)  // A 客场进球
                val awayGoalsB = (leg2?.awayScoreSim ?: 0)  // B 客场进球
                when {
                    awayGoalsB > awayGoalsA -> tie.homeClubId!!
                    awayGoalsA > awayGoalsB -> tie.awayClubId!!
                    else -> if (config.cupUsePenaltyShootout) {
                        if (Random.nextBoolean()) tie.homeClubId!! else tie.awayClubId!!
                    } else -1
                }
            }
        }
    }
}
```

## 九、性能要求

| 操作 | 目标 | 实现策略 |
|------|------|----------|
| 赛程生成（完整赛季 380+ 场） | ≤500ms | 圆盘旋转法 O(n²) + 批量事务插入 + 后台协程 |
| 积分榜更新（单场结束后） | ≤100ms | 增量累加 + 单赛事重排 + 缓存快照 |
| 赛程页列表加载 | ≤200ms | 分页 50 条 + 索引 (saveId, matchDate) |
| 积分榜查询 | ≤50ms | cache.db 优先读，TTL 5 分钟 |
| 杯赛对阵树渲染 | ≤100ms | LazyColumn 分阶段渲染，非全量绘制 |

性能优化要点：
1. **赛程生成后台化**：`generateSeasonSchedule` 在 IO 协程执行，UI 显示进度。
2. **积分榜增量更新**：单场结束后只重算受影响赛事，非全赛季。
3. **批量事务**：赛程插入用 `runInTransaction`，避免逐条提交。
4. **索引完备**：`save_match(saveId, matchDate)` / `save_league_table(saveId, competitionId, rank)` 必须建索引。
5. **缓存快照**：积分榜写入 cache.db JSON 快照，查询优先读缓存。

## 十四、关键风险与应对

### 1. 赛程生成主客不平衡

- **表现**：某队主场数 ≠ 客场数，或连续 4+ 主场
- **应对**：
  - 圆盘旋转法保证基础平衡（第二半程主客对换）
  - adjustConsecutiveHomeAway 多次尝试调整
  - 单测强制校验主客数相等 + 连续 ≤3

### 2. 杯赛对阵树回填错误

- **表现**：晋级方未正确进入下一轮，或对阵树断裂
- **应对**：
  - nextTieId 显式记录晋级路径
  - resolveCupTie 自动回填 + 双方就绪后生成比赛
  - 单测覆盖完整杯赛流程（32→决赛）

### 3. 积分榜同分排名歧义

- **表现**：相互战绩未踢完或同分多队循环
- **应对**：
  - 配置化 tiebreaker_order，按序比较
  - 相互战绩只比较已踢比赛，未踢按 0 处理
  - 全相同时稳定排序（保持原顺序），可后期加附加赛

### 4. 性能不达标

- **表现**：赛程生成 >500ms 或积分榜更新 >100ms
- **应对**：
  - 批量事务插入，避免逐条
  - 积分榜增量更新（单赛事重算，非全赛季）
  - cache.db 快照缓存，TTL 5 分钟
  - 后台协程执行，UI 显示进度

### 5. 国际比赛日与俱乐部赛程冲突

- **表现**：国脚被征召期间俱乐部仍有比赛
- **应对**：
  - blackoutDates 黑名单机制，该期间不排俱乐部比赛
  - redistributeDates 自动跳过黑名单日
  - 征召期间球员 isReleased 标记，俱乐部比赛不可用
